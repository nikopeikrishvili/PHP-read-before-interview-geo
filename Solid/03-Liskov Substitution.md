**Liskov Substitution Principle (LSP)** рЃљрЃарЃўрЃА SOLID рЃърЃарЃўрЃюрЃфрЃўрЃърЃћрЃЉрЃўрЃА рЃћрЃарЃЌ-рЃћрЃарЃЌрЃў рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЋрЃљрЃюрЃў рЃюрЃљрЃгрЃўрЃџрЃў, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃќрЃћ рЃЮрЃарЃўрЃћрЃюрЃбрЃўрЃарЃћрЃЉрЃБрЃџ рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃўрЃарЃћрЃЉрЃљрЃерЃў (OOP) рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћрЃЉрЃљ. рЃўрЃА рЃњрЃЋрЃЎрЃљрЃарЃюрЃљрЃ«рЃЮрЃЉрЃА рЃарЃЮрЃЏ **рЃЏрЃерЃЮрЃЉрЃћрЃџрЃў рЃЎрЃџрЃљрЃАрЃўрЃА рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃў рЃюрЃћрЃЉрЃўрЃА рЃЏрЃўрЃћрЃа рЃЊрЃарЃЮрЃА рЃБрЃюрЃЊрЃљ рЃерЃћрЃўрЃфрЃЋрЃљрЃџрЃЮрЃА рЃЏрЃљрЃЌрЃў рЃерЃЋрЃўрЃџрЃЮрЃЉрЃўрЃџрЃў рЃЎрЃџрЃљрЃАрЃўрЃА рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃЌ рЃўрЃАрЃћ, рЃарЃЮрЃЏ рЃљрЃЏрЃљрЃю рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљрЃерЃў рЃерЃћрЃфрЃЊрЃЮрЃЏрЃћрЃЉрЃў рЃљрЃа рЃњрЃљрЃЏрЃЮрЃўрЃгрЃЋрЃўрЃЮрЃА.

рЃћрЃА рЃърЃарЃўрЃюрЃфрЃўрЃърЃў рЃ«рЃерЃўрЃарЃљрЃЊ рЃњрЃЋрЃћрЃ«рЃЏрЃљрЃарЃћрЃЉрЃљ рЃЏрЃћрЃЏрЃЎрЃЋрЃўрЃЊрЃарЃћрЃЮрЃЉрЃўрЃА рЃАрЃгрЃЮрЃарЃў рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃўрЃА рЃЊрЃарЃЮрЃА, рЃарЃљрЃЌрЃљ рЃЌрЃљрЃЋрЃўрЃЊрЃљрЃю рЃљрЃЋрЃўрЃфрЃўрЃџрЃЮрЃЌ рЃўрЃАрЃћрЃЌрЃў рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃћрЃЉрЃў, рЃарЃЮрЃЊрЃћрЃћрЃАрЃљрЃф рЃерЃЋрЃўрЃџрЃЎрЃЮрЃЉрЃўрЃџрЃў рЃЎрЃџрЃљрЃАрЃў рЃљрЃарЃдрЃЋрЃћрЃЋрЃА рЃЏрЃерЃЮрЃЉрЃћрЃџрЃў рЃЎрЃџрЃљрЃАрЃўрЃА рЃЦрЃфрЃћрЃЋрЃљрЃА.

---

## 1. Liskov Substitution Principle- рЃўрЃА рЃљрЃ«рЃАрЃюрЃљ

рЃџрЃўрЃАрЃЎрЃЮрЃЋрЃўрЃА рЃЕрЃљрЃюрЃљрЃфрЃЋрЃџрЃћрЃЉрЃўрЃА рЃърЃарЃўрЃюрЃфрЃўрЃърЃў рЃњрЃЋрЃљрЃЋрЃљрЃџрЃЊрЃћрЃЉрЃБрЃџрЃћрЃЉрЃА, рЃарЃЮрЃЏ **рЃЌрЃБ B рЃЎрЃџрЃљрЃАрЃў рЃЏрЃћрЃЏрЃћрЃЎрЃЋрЃўрЃЊрЃарЃћрЃЮрЃЉрЃўрЃЌ рЃўрЃдрЃћрЃЉрЃА A рЃЎрЃџрЃљрЃАрЃА (A->B), рЃЏрЃљрЃерЃўрЃю B-рЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ рЃБрЃюрЃЊрЃљ рЃўрЃДрЃЮрЃА рЃерЃћрЃАрЃљрЃФрЃџрЃћрЃЉрЃћрЃџрЃў A-рЃА рЃЏрЃљрЃњрЃўрЃЋрЃарЃљрЃЊ, рЃўрЃАрЃћ рЃарЃЮрЃЏ рЃљрЃърЃџрЃўрЃЎрЃљрЃфрЃўрЃўрЃА рЃЦрЃБрЃюрЃфрЃўрЃЮрЃюрЃўрЃарЃћрЃЉрЃљ рЃљрЃа рЃерЃћрЃўрЃфрЃЋрЃљрЃџрЃЮрЃА.**

рЃЌрЃБ рЃерЃЋрЃўрЃџрЃЮрЃЉрЃўрЃџрЃў рЃЎрЃџрЃљрЃАрЃў рЃфрЃЋрЃџрЃўрЃА рЃЏрЃерЃЮрЃЉрЃћрЃџрЃў рЃЎрЃџрЃљрЃАрЃўрЃА рЃЦрЃфрЃћрЃЋрЃљрЃА рЃўрЃАрЃћ, рЃарЃЮрЃЏ рЃЎрЃЮрЃЊрЃў, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃЏрЃерЃЮрЃЉрЃћрЃџ рЃЎрЃџрЃљрЃАрЃА рЃўрЃДрЃћрЃюрЃћрЃЉрЃЊрЃљ, рЃљрЃарЃљрЃАрЃгрЃЮрЃарЃљрЃЊ рЃўрЃгрЃДрЃћрЃЉрЃА рЃЏрЃБрЃерЃљрЃЮрЃЉрЃљрЃА, рЃЏрЃљрЃерЃўрЃю LSP рЃЊрЃљрЃўрЃарЃдрЃЋрЃљ.

---

## 2. LSP рЃЊрЃљрЃарЃдрЃЋрЃћрЃЋрЃўрЃА рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў

**РЮї рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў:**

1. **рЃЏрЃерЃЮрЃЉрЃћрЃџрЃў рЃЎрЃџрЃљрЃАрЃў:**
```php
class Rectangle {
    protected int $width;
    protected int $height;

    public function setWidth(int $width): void {
        $this->width = $width;
    }

    public function setHeight(int $height): void {
        $this->height = $height;
    }

    public function getArea(): int {
        return $this->width * $this->height;
    }
}
```

рЃљрЃЏ рЃЎрЃџрЃљрЃАрЃўрЃА рЃџрЃЮрЃњрЃўрЃЎрЃљ рЃАрЃгрЃЮрЃарЃўрЃљ рЃЊрЃљ рЃЏрЃўрЃАрЃњрЃљрЃю рЃњрЃљрЃЏрЃЮрЃЏрЃЊрЃўрЃюрЃљрЃарЃћ рЃЎрЃџрЃљрЃАрЃћрЃЉрЃў рЃБрЃюрЃЊрЃљ рЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЮрЃЊрЃћрЃА рЃЏрЃўрЃА рЃЦрЃфрЃћрЃЋрЃљрЃА.

2. **рЃЦрЃЋрЃћрЃЎрЃџрЃљрЃАрЃў, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃљрЃарЃЕрЃЋрЃћрЃЋрЃА LSP-рЃА**
```php
class Square extends Rectangle {
    public function setWidth(int $width): void {
        $this->width = $width;
        $this->height = $width; // рЃЎрЃЋрЃљрЃЊрЃарЃљрЃбрЃў рЃАрЃўрЃњрЃљрЃюрЃћрЃА рЃЊрЃљ рЃАрЃўрЃЏрЃљрЃдрЃџрЃћрЃА рЃћрЃарЃЌрЃюрЃљрЃўрЃарЃљрЃЊ рЃфрЃЋрЃџрЃўрЃА
    }

    public function setHeight(int $height): void {
        $this->height = $height;
        $this->width = $height; // рЃЎрЃЋрЃљрЃЊрЃарЃљрЃбрЃў рЃАрЃўрЃЏрЃљрЃдрЃџрЃћрЃА рЃЊрЃљ рЃАрЃўрЃњрЃљрЃюрЃћрЃА рЃћрЃарЃЌрЃюрЃљрЃўрЃарЃљрЃЊ рЃфрЃЋрЃџрЃўрЃА
    }
}
```

3. **LSP-рЃА рЃЊрЃљрЃарЃдрЃЋрЃћрЃЋрЃўрЃА рЃбрЃћрЃАрЃбрЃў**
```code
function processRectangle(Rectangle $rectangle) {
    $rectangle->setWidth(5);
    $rectangle->setHeight(10);

    if ($rectangle->getArea() !== 50) {
        throw new Exception("Liskov Substitution Principle is violated!");
    }

    echo "Area: " . $rectangle->getArea() . PHP_EOL;
}

$rectangle = new Rectangle();
processRectangle($rectangle); // OK: Area: 50

$square = new Square();
processRectangle($square); // рЃерЃћрЃфрЃЊрЃЮрЃЏрЃљ! LSP рЃЊрЃљрЃўрЃарЃдрЃЋрЃљ
```

**рЃърЃарЃЮрЃЉрЃџрЃћрЃЏрЃљ:**

рЃЌрЃБ **Square** рЃЎрЃџрЃљрЃАрЃўрЃА рЃЮрЃЉрЃўрЃћрЃбрЃА рЃњрЃљрЃЊрЃљрЃЋрЃфрЃћрЃЏрЃЌ **processRectangle()** рЃцрЃБрЃюрЃЦрЃфрЃўрЃљрЃА, рЃерЃћрЃЊрЃћрЃњрЃў рЃљрЃарЃљрЃАрЃгрЃЮрЃарЃў рЃњрЃљрЃЏрЃЮрЃЋрЃљ, рЃарЃљрЃЊрЃњрЃљрЃю **setWidth()** рЃљрЃю **setHeight()** рЃЮрЃарЃўрЃЋрЃћ рЃърЃљрЃарЃљрЃЏрЃћрЃбрЃА рЃфрЃЋрЃџрЃўрЃА, рЃарЃљрЃф **Rectangle** рЃЎрЃџрЃљрЃАрЃерЃў рЃЊрЃљрЃерЃЋрЃћрЃЉрЃБрЃџрЃў рЃљрЃа рЃљрЃарЃўрЃА.

---
## 3. Liskov Substitution Principle - рЃўрЃА рЃАрЃгрЃЮрЃарЃљрЃЊ рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ

рЃљрЃЏ рЃърЃарЃЮрЃЉрЃџрЃћрЃЏрЃўрЃА рЃњрЃљрЃЊрЃљрЃАрЃљрЃГрЃарЃћрЃџрЃљрЃЊ, рЃерЃћрЃњрЃЋрЃўрЃФрЃџрЃўрЃљ рЃЏрЃЮрЃЋрЃљрЃ«рЃЊрЃўрЃюрЃЮрЃЌ рЃарЃћрЃцрЃљрЃЦрЃбрЃЮрЃарЃўрЃюрЃњрЃў рЃЊрЃљ рЃЎрЃџрЃљрЃАрЃћрЃЉрЃўрЃА рЃЏрЃћрЃЏрЃЎрЃЋрЃўрЃЊрЃарЃћрЃЮрЃЉрЃљ рЃАрЃгрЃЮрЃарЃћрЃЊ рЃњрЃљрЃюрЃЋрЃАрЃљрЃќрЃдрЃЋрЃарЃЮрЃЌ.
рЃљрЃЏ рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃљрЃерЃў, **Rectangle** рЃЊрЃљ **Square** рЃЎрЃџрЃљрЃАрЃћрЃЉрЃў рЃњрЃљрЃюрЃАрЃ«рЃЋрЃљрЃЋрЃћрЃЉрЃБрЃџрЃў рЃБрЃюрЃЊрЃљ рЃўрЃДрЃЮрЃА рЃЊрЃљ рЃљрЃа рЃБрЃюрЃЊрЃљ рЃўрЃДрЃЋрЃюрЃћрЃю рЃћрЃарЃЌрЃЏрЃљрЃюрЃћрЃЌрЃЌрЃљрЃю **[[02-Inheritance (рЃЏрЃћрЃЏрЃЎрЃЋрЃўрЃЊрЃарЃћрЃЮрЃЉрЃљ)]]** рЃўрЃЌ рЃЊрЃљрЃЎрЃљрЃЋрЃерЃўрЃарЃћрЃЉрЃБрЃџрЃћрЃЉрЃў.

```php
interface Shape {
    public function getArea(): int;
}

class Rectangle implements Shape {
    protected int $width;
    protected int $height;

    public function __construct(int $width, int $height) {
        $this->width = $width;
        $this->height = $height;
    }

    public function getArea(): int {
        return $this->width * $this->height;
    }
}

class Square implements Shape {
    protected int $side;

    public function __construct(int $side) {
        $this->side = $side;
    }

    public function getArea(): int {
        return $this->side * $this->side;
    }
}

function processShape(Shape $shape) {
    echo "Area: " . $shape->getArea() . PHP_EOL;
}

$rectangle = new Rectangle(5, 10);
$square = new Square(5);

processShape($rectangle); // Area: 50
processShape($square);    // Area: 25
```

**рЃњрЃљрЃЊрЃљрЃгрЃДрЃЋрЃћрЃбрЃљ:**

РюЁ рЃерЃћрЃЋрЃЦрЃЏрЃћрЃюрЃўрЃЌ **Shape** рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃў, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃљрЃдрЃгрЃћрЃарЃА **getArea()** рЃЏрЃћрЃЌрЃЮрЃЊрЃА.
РюЁ **Rectangle** рЃЊрЃљ **Square** рЃЊрЃљрЃЏрЃЮрЃБрЃЎрЃўрЃЊрЃћрЃЉрЃџрЃљрЃЊ рЃљрЃ«рЃЊрЃћрЃюрЃћрЃю рЃљрЃЏ рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃўрЃА рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃљрЃА, рЃЊрЃљ рЃћрЃарЃЌрЃЏрЃљрЃюрЃћрЃЌрЃЌрЃљрЃю рЃљрЃа рЃЏрЃћрЃЏрЃЎрЃЋрЃўрЃЊрЃарЃћрЃЮрЃЉрЃћрЃю.
РюЁ **processShape()** рЃцрЃБрЃюрЃЦрЃфрЃўрЃљ рЃЏрЃБрЃерЃљрЃЮрЃЉрЃА рЃАрЃгрЃЮрЃарЃљрЃЊ, рЃарЃљрЃЊрЃњрЃљрЃю **LSP** рЃЊрЃљрЃфрЃБрЃџрЃўрЃљ

**­Ъџђ рЃЊрЃљрЃАрЃЎрЃЋрЃюрЃљ**

**Liskov Substitution Principle** - рЃўрЃА рЃАрЃгрЃЮрЃарЃћрЃЊ рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ  рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃљрЃА рЃњрЃЋрЃљрЃФрЃџрЃћрЃЋрЃА рЃљрЃЋрЃўрЃарЃўрЃЊрЃЮрЃЌ рЃЏрЃћрЃЏрЃЎрЃЋрЃўрЃЊрЃарЃћрЃЮрЃЉрЃўрЃА рЃљрЃарЃљрЃАрЃгрЃЮрЃарЃў рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃћрЃЉрЃў рЃЊрЃљ рЃњрЃљрЃЋрЃ«рЃљрЃЊрЃЮрЃЌ рЃЕрЃЋрЃћрЃюрЃў рЃЎрЃЮрЃЊрЃў рЃБрЃцрЃарЃЮ рЃњрЃљрЃАрЃљрЃњрЃћрЃЉрЃў, рЃЏрЃљрЃАрЃерЃбрЃљрЃЉрЃўрЃарЃћрЃЉрЃљрЃЊрЃў рЃЊрЃљ рЃАрЃбрЃљрЃЉрЃўрЃџрЃБрЃарЃў.

**рЃљрЃЏ рЃърЃарЃўрЃюрЃфрЃўрЃърЃћрЃЉрЃўрЃА рЃЊрЃљрЃарЃдрЃЋрЃћрЃЋрЃљрЃЏ рЃерЃћрЃўрЃФрЃџрЃћрЃЉрЃљ рЃњрЃљрЃЏрЃЮрЃўрЃгрЃЋрЃўрЃЮрЃА:**
**РЮї рЃЏрЃЮрЃБрЃџрЃЮрЃЊрЃюрЃћрЃџрЃў рЃерЃћрЃфрЃЊрЃЮрЃЏрЃћрЃЉрЃў, рЃарЃљрЃЊрЃњрЃљрЃюрЃљрЃф рЃЏрЃерЃЮрЃЉрЃћрЃџрЃў рЃЊрЃљ рЃерЃЋрЃўрЃџрЃЮрЃЉрЃўрЃџрЃў рЃЎрЃџрЃљрЃАрЃћрЃЉрЃў рЃћрЃарЃЌрЃЏрЃљрЃюрЃћрЃЌрЃА рЃЦрЃфрЃћрЃЋрЃћрЃЉрЃерЃў рЃљрЃа рЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃљ**.    
**РЮї рЃарЃЌрЃБрЃџрЃљрЃЊ рЃњрЃљрЃАрЃљрЃњрЃћрЃЉрЃў рЃЊрЃљ рЃарЃЌрЃБрЃџрЃљрЃЊ рЃњрЃљрЃАрЃгрЃЮрЃарЃћрЃЉрЃљрЃЊрЃў рЃЎрЃЮрЃЊрЃў, рЃарЃљрЃЏрЃљрЃф рЃерЃћрЃўрЃФрЃџрЃћрЃЉрЃљ рЃ«рЃћрЃџрЃў рЃерЃћрЃБрЃерЃљрЃџрЃЮрЃА рЃЎрЃЮрЃЊрЃўрЃА рЃњрЃљрЃЏрЃљрЃарЃЌрЃБрЃџ рЃЏрЃБрЃерЃљрЃЮрЃЉрЃљрЃА рЃЊрЃљ рЃбрЃћрЃАрЃбрЃўрЃарЃћрЃЉрЃљрЃА.**    

­ЪњА **рЃърЃарЃљрЃЦрЃбрЃўрЃЎрЃБрЃџрЃў рЃарЃЕрЃћрЃЋрЃљ:**
рЃЌрЃБ рЃ«рЃћрЃЊрЃљрЃЋрЃЌ, рЃарЃЮрЃЏ рЃерЃЋрЃўрЃџрЃЮрЃЉрЃўрЃџрЃў рЃЎрЃџрЃљрЃАрЃў рЃљрЃарЃдрЃЋрЃћрЃЋрЃА рЃЏрЃерЃЮрЃЉрЃџрЃўрЃА рЃЦрЃфрЃћрЃЋрЃљрЃА, рЃ»рЃЮрЃЉрЃўрЃљ рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃЮрЃЌ рЃЎрЃЮрЃЏрЃърЃЮрЃќрЃўрЃфрЃўрЃљ рЃЏрЃћрЃЏрЃЎрЃЋрЃўрЃЊрЃарЃћрЃЮрЃЉрЃўрЃА рЃюрЃљрЃфрЃЋрЃџрЃљрЃЊ рЃЊрЃљ рЃњрЃљрЃљрЃюрЃљрЃгрЃўрЃџрЃЮрЃЌ рЃцрЃБрЃюрЃЦрЃфрЃўрЃЮрЃюрЃљрЃџрЃў рЃАрЃгрЃЮрЃарЃў рЃњрЃќрЃўрЃЌ.
