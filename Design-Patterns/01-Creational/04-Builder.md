**Builder** рЃЏрЃерЃћрЃюрЃћрЃЉрЃћрЃџрЃў рЃљрЃарЃўрЃА Creational рЃбрЃўрЃърЃўрЃА рЃЊрЃўрЃќрЃљрЃўрЃюрЃўрЃА рЃърЃљрЃбрЃћрЃарЃюрЃў, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћрЃЉрЃљ рЃарЃЌрЃБрЃџрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃЎрЃЮрЃюрЃАрЃбрЃарЃБрЃфрЃўрЃўрЃА рЃњрЃљрЃЏрЃљрЃарЃбрЃўрЃЋрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА. рЃЏрЃўрЃАрЃў рЃЏрЃЌрЃљрЃЋрЃљрЃарЃў рЃўрЃЊрЃћрЃљрЃљ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃА рЃљрЃњрЃћрЃЉрЃљ рЃћрЃбрЃљрЃърЃЮрЃЉрЃарЃўрЃЋрЃљрЃЊ, рЃЎрЃЮрЃЏрЃърЃџрЃћрЃЦрЃАрЃБрЃарЃў рЃЎрЃЮрЃюрЃАрЃбрЃарЃўрЃбрЃЮрЃЮрЃарЃћрЃЉрЃўрЃА рЃЌрЃљрЃЋрЃўрЃЊрЃљрЃю рЃљрЃАрЃљрЃфрЃўрЃџрЃћрЃЉрЃџрЃљрЃЊ.

Builder рЃърЃљрЃбрЃћрЃарЃюрЃў рЃњрЃљрЃюрЃАрЃљрЃЎрЃБрЃЌрЃарЃћрЃЉрЃўрЃЌ рЃАрЃљрЃАрЃљрЃарЃњрЃћрЃЉрЃџрЃЮрЃљрЃЊ, рЃарЃЮрЃфрЃљ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃА рЃЉрЃћрЃЋрЃарЃў рЃърЃљрЃарЃљрЃЏрЃћрЃбрЃарЃў рЃљрЃЦрЃЋрЃА рЃЊрЃљ рЃЏрЃљрЃЌрЃў рЃЎрЃЮрЃЏрЃЉрЃўрЃюрЃљрЃфрЃўрЃћрЃЉрЃў рЃАрЃ«рЃЋрЃљрЃЊрЃљрЃАрЃ«рЃЋрЃљ рЃЋрЃљрЃарЃўрЃљрЃюрЃбрЃћрЃЉрЃерЃў рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћрЃЉрЃљ.

---
##### рЃарЃЮрЃЊрЃўрЃА рЃБрЃюрЃЊрЃљ рЃњрЃљрЃЏрЃЮрЃЋрЃўрЃДрЃћрЃюрЃЮрЃЌ Builder?
- рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃў рЃарЃЌрЃБрЃџрЃў рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃўрЃАрЃљрЃљ рЃЊрЃљ рЃЏрЃўрЃАрЃў рЃљрЃњрЃћрЃЉрЃљ рЃЏрЃЮрЃўрЃЌрЃ«рЃЮрЃЋрЃА рЃарЃљрЃЏрЃЊрЃћрЃюрЃўрЃЏрЃћ рЃюрЃљрЃЉрЃўрЃ»рЃА.
- рЃАрЃљрЃГрЃўрЃарЃЮрЃљ рЃАрЃ«рЃЋрЃљрЃЊрЃљрЃАрЃ«рЃЋрЃљ рЃЉрЃћрЃарЃАрЃўрЃўрЃЌ рЃљрЃњрЃћрЃЉрЃБрЃџрЃў рЃћрЃарЃЌрЃў рЃЊрЃљ рЃўрЃњрЃўрЃЋрЃћ рЃбрЃўрЃърЃўрЃА рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ.
- рЃЎрЃЮрЃюрЃАрЃбрЃарЃБрЃЦрЃбрЃЮрЃарЃў рЃФрЃљрЃџрЃўрЃљрЃю рЃњрЃарЃФрЃћрЃџрЃўрЃљ рЃЊрЃљ рЃЎрЃЮрЃЊрЃўрЃА рЃгрЃљрЃЎрЃўрЃЌрЃ«рЃЋрЃљ рЃарЃЌрЃБрЃџрЃЊрЃћрЃЉрЃљ.

---

##### Builder рЃърЃљрЃбрЃћрЃарЃюрЃўрЃА рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃљ

рЃерЃћрЃЋрЃЦрЃЏрЃюрЃљрЃЌ **Car** рЃЎрЃџрЃљрЃАрЃўрЃА рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃў **Builder** рЃърЃљрЃбрЃћрЃарЃюрЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃўрЃЌ.

###### рЃърЃарЃЮрЃЊрЃБрЃЦрЃбрЃўрЃА рЃЎрЃџрЃљрЃАрЃў (Product)

рЃћрЃА рЃљрЃарЃўрЃА рЃЎрЃџрЃљрЃАрЃў рЃарЃЮрЃЏрЃџрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљрЃф рЃњрЃЋрЃўрЃюрЃЊрЃљ:
```php
class Car {
    public string $engine;
    public int $seats;
    public bool $gps;
    
    public function __construct(string $engine, int $seats, bool $gps) {
        $this->engine = $engine;
        $this->seats = $seats;
        $this->gps = $gps;
    }

    public function showSpecifications(): void {
        echo "Engine: {$this->engine}, Seats: {$this->seats}, GPS: " . ($this->gps ? 'Yes' : 'No') . PHP_EOL;
    }
}
```

###### Builder рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃў
рЃћрЃА рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃў рЃњрЃљрЃюрЃАрЃљрЃќрЃдрЃЋрЃарЃљрЃЋрЃА, рЃарЃљ рЃЏрЃћрЃЌрЃЮрЃЊрЃћрЃЉрЃў рЃБрЃюрЃЊрЃљ рЃцрЃЦрЃЮрЃюрЃЊрЃћрЃА рЃДрЃЋрЃћрЃџрЃљ рЃЉрЃўрЃџрЃЊрЃћрЃарЃА.

```php
interface CarBuilder {
    public function setEngine(string $type): CarBuilder;
    public function setSeats(int $count): CarBuilder;
    public function setGPS(bool $hasGPS): CarBuilder;
    public function build(): Car;
}
```

###### рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃарЃџрЃў рЃЉрЃўрЃџрЃЊрЃћрЃарЃў (Concrete Builder)

рЃћрЃА рЃЎрЃџрЃљрЃАрЃў рЃБрЃерЃБрЃљрЃџрЃЮрЃЊ рЃљрЃерЃћрЃюрЃћрЃЉрЃА **Car** рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃА.

```php
class ConcreteCarBuilder implements CarBuilder {
    private string $engine;
    private int $seats;
    private bool $gps;

    public function setEngine(string $type): CarBuilder {
        $this->engine = $type;
        return $this;
    }

    public function setSeats(int $count): CarBuilder {
        $this->seats = $count;
        return $this;
    }

    public function setGPS(bool $hasGPS): CarBuilder {
        $this->gps = $hasGPS;
        return $this;
    }

    public function build(): Car {
        return new Car($this->engine, $this->seats, $this->gps);
    }
}
```

###### Director (рЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃў)

**Director** рЃЏрЃљрЃарЃЌрЃљрЃЋрЃА рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃўрЃА рЃърЃарЃЮрЃфрЃћрЃАрЃА. рЃЏрЃўрЃАрЃў рЃЏрЃўрЃќрЃљрЃюрЃўрЃљ **Builder**-рЃўрЃА рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃўрЃЌ рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃАрЃбрЃљрЃюрЃЊрЃљрЃарЃбрЃБрЃџрЃў рЃЋрЃћрЃарЃАрЃўрЃћрЃЉрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ.

```php
class CarDirector {
    public function buildSportsCar(CarBuilder $builder): Car {
        return $builder
            ->setEngine('V8')
            ->setSeats(2)
            ->setGPS(true)
            ->build();
    }

    public function buildFamilyCar(CarBuilder $builder): Car {
        return $builder
            ->setEngine('V6')
            ->setSeats(5)
            ->setGPS(false)
            ->build();
    }
}
```

###### рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ (рЃЎрЃџрЃўрЃћрЃюрЃбрЃўрЃА рЃЎрЃЮрЃЊрЃў)

```php
$builder = new ConcreteCarBuilder();
$director = new CarDirector();

$sportsCar = $director->buildSportsCar($builder);
$familyCar = $director->buildFamilyCar($builder);

$sportsCar->showSpecifications(); // Engine: V8, Seats: 2, GPS: Yes
$familyCar->showSpecifications(); // Engine: V6, Seats: 5, GPS: No
```

**­Ъџђ рЃЊрЃљрЃАрЃЎрЃЋрЃюрЃљ**

**Builder рЃърЃљрЃбрЃћрЃарЃюрЃў** рЃњрЃЋрЃљрЃФрЃџрЃћрЃЋрЃА рЃЏрЃЮрЃЦрЃюрЃўрЃџрЃЮрЃЉрЃљрЃА рЃарЃЌрЃБрЃџрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃўрЃАрЃљрЃА. рЃЏрЃўрЃАрЃў рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ рЃњрЃљрЃюрЃАрЃљрЃЎрЃБрЃЌрЃћрЃЉрЃўрЃЌ рЃАрЃљрЃАрЃљрЃарЃњрЃћрЃЉрЃџрЃЮрЃљрЃЊ рЃЏрЃљрЃерЃўрЃю, рЃарЃЮрЃфрЃљ рЃњрЃЋрЃљрЃЦрЃЋрЃА **рЃЏрЃарЃљрЃЋрЃљрЃџрЃцрЃљрЃќрЃўрЃљрЃюрЃў рЃЎрЃЮрЃюрЃАрЃбрЃарЃБрЃЦрЃфрЃўрЃљ** рЃљрЃю **рЃЉрЃћрЃЋрЃарЃў рЃърЃљрЃарЃљрЃЏрЃћрЃбрЃарЃў**, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃерЃћрЃўрЃФрЃџрЃћрЃЉрЃљ рЃАрЃ«рЃЋрЃљрЃЊрЃљрЃАрЃ«рЃЋрЃљ рЃњрЃќрЃўрЃЌ рЃўрЃДрЃЮрЃА рЃЎрЃЮрЃюрЃцрЃўрЃњрЃБрЃарЃўрЃарЃћрЃЉрЃБрЃџрЃў.

**Builder** - рЃўрЃА рЃЏрЃЌрЃљрЃЋрЃљрЃарЃў рЃБрЃърЃўрЃарЃљрЃбрЃћрЃАрЃЮрЃЉрЃћрЃЉрЃў:

РюЁ рЃЎрЃЮрЃЊрЃўрЃА рЃБрЃЎрЃћрЃЌрЃћрЃАрЃў рЃЮрЃарЃњрЃљрЃюрЃўрЃќрЃћрЃЉрЃљ.     
РюЁ рЃЏрЃљрЃарЃбрЃўрЃЋрЃў рЃњрЃљрЃцрЃљрЃарЃЌрЃЮрЃћрЃЉрЃљ.    
РюЁ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃЎрЃЮрЃюрЃцрЃўрЃњрЃБрЃарЃљрЃфрЃўрЃўрЃА рЃњрЃљрЃЏрЃљрЃарЃбрЃўрЃЋрЃћрЃЉрЃљ.
