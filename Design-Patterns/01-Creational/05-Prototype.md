**Prototype** рЃљрЃарЃўрЃА рЃљрЃарЃЌ-рЃћрЃарЃЌрЃў Creational рЃбрЃўрЃърЃўрЃА рЃЊрЃўрЃќрЃљрЃўрЃю рЃърЃљрЃбрЃћрЃарЃюрЃў, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћрЃЉрЃљ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃЎрЃџрЃЮрЃюрЃўрЃарЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА, рЃарЃљрЃф рЃљрЃЏрЃфрЃўрЃарЃћрЃЉрЃА **new** рЃЮрЃърЃћрЃарЃљрЃбрЃЮрЃарЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљрЃА рЃЊрЃљ рЃљрЃЏрЃљрЃарЃбрЃўрЃЋрЃћрЃЉрЃА рЃљрЃ«рЃљрЃџрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљрЃА.

**Prototype**-рЃўрЃА рЃБрЃърЃўрЃарЃљрЃбрЃћрЃАрЃЮрЃЉрЃљ рЃўрЃА рЃљрЃарЃўрЃА, рЃарЃЮрЃЏ рЃўрЃА рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃљрЃА рЃњрЃЋрЃљрЃФрЃџрЃћрЃЋрЃА, рЃерЃћрЃЋрЃЦрЃЏрЃюрЃљрЃЌ рЃљрЃ«рЃљрЃџрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃў рЃљрЃарЃАрЃћрЃЉрЃБрЃџрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃА рЃАрЃљрЃцрЃБрЃФрЃЋрЃћрЃџрЃќрЃћ, рЃЎрЃџрЃЮрЃюрЃўрЃарЃћрЃЉрЃўрЃА рЃњрЃќрЃўрЃЌ. рЃћрЃА рЃњрЃљрЃюрЃАрЃљрЃЎрЃБрЃЌрЃарЃћрЃЉрЃўрЃЌ рЃАрЃљрЃАрЃљрЃарЃњрЃћрЃЉрЃџрЃЮрЃљрЃЊ рЃЏрЃљрЃерЃўрЃю, рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ рЃарЃЌрЃБрЃџрЃўрЃљ рЃљрЃю рЃЏрЃљрЃАрЃќрЃћ рЃЉрЃћрЃЋрЃарЃў рЃарЃћрЃАрЃБрЃарЃАрЃў рЃўрЃ«рЃљрЃарЃ»рЃћрЃЉрЃљ.

---
##### Prototype рЃЊрЃўрЃќрЃљрЃўрЃю рЃърЃљрЃбрЃћрЃарЃюрЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ.

PHP-рЃерЃў Prototype рЃърЃљрЃбрЃћрЃарЃюрЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃюрЃўрЃАрЃЌрЃЋрЃўрЃА рЃљрЃБрЃфрЃўрЃџрЃћрЃЉрЃћрЃџрЃўрЃљ cloneрЃцрЃБрЃюрЃЦрЃфрЃўрЃўрЃА рЃњрЃљрЃЏрЃЮрЃФрЃљрЃ«рЃћрЃЉрЃљ. рЃЏрЃЮрЃЊрЃў, рЃњрЃљрЃюрЃЋрЃўрЃ«рЃўрЃџрЃЮрЃЌ рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў.

```php
<?php

// рЃљрЃЉрЃАрЃбрЃарЃљрЃЦрЃбрЃБрЃџрЃў рЃЎрЃџрЃљрЃАрЃў, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф Prototype-рЃА рЃњрЃљрЃюрЃАрЃљрЃќрЃдрЃЋрЃарЃљрЃЋрЃА
abstract class Shape
{
    public $color;

    public function setColor($color)
    {
        $this->color = $color;
    }

    // Prototype рЃЏрЃћрЃЌрЃЮрЃЊрЃў - рЃЎрЃџрЃЮрЃюрЃўрЃарЃћрЃЉрЃљ
    abstract public function __clone();
}

// рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџрЃў рЃЎрЃџрЃљрЃАрЃћрЃЉрЃў, рЃарЃЮрЃЏрЃџрЃћрЃЉрЃўрЃф Prototype-рЃА рЃўрЃДрЃћрЃюрЃћрЃЉрЃћрЃю
class Circle extends Shape
{
    public $radius;

    public function __construct($radius)
    {
        $this->radius = $radius;
    }

    public function __clone()
    {
        // рЃљрЃЦ рЃерЃћрЃњрЃЋрЃўрЃФрЃџрЃўрЃљ рЃАрЃљрЃГрЃўрЃарЃЮрЃћрЃЉрЃўрЃАрЃљрЃЏрЃћрЃЉрЃа рЃЊрЃљрЃЋрЃљрЃЏрЃљрЃбрЃЮрЃЌ рЃџрЃЮрЃњрЃўрЃЎрЃљ
    }
}

// Prototype-рЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ
$circle1 = new Circle(10);
$circle1->setColor('Red');

$circle2 = clone $circle1; // рЃљрЃ«рЃљрЃџрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ рЃЎрЃџрЃЮрЃюрЃўрЃарЃћрЃЉрЃўрЃЌ
$circle2->setColor('Blue'); // рЃцрЃћрЃарЃўрЃА рЃерЃћрЃфрЃЋрЃџрЃљ рЃљрЃ«рЃљрЃџрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃАрЃЌрЃЋрЃўрЃА

echo "Circle1 Color: " . $circle1->color . "\n"; // Red
echo "Circle2 Color: " . $circle2->color . "\n"; // Blue

?>
```

##### рЃарЃљ рЃБрЃърЃўрЃарЃљрЃбрЃћрЃАрЃЮрЃЉрЃћрЃЉрЃў рЃљрЃЦрЃЋрЃА Prototype рЃърЃљрЃбрЃћрЃарЃюрЃА?
1. **рЃерЃћрЃАрЃарЃБрЃџрЃћрЃЉрЃўрЃА рЃАрЃўрЃЕрЃЦрЃљрЃарЃћ** - рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ рЃБрЃцрЃарЃЮ рЃАрЃгрЃарЃљрЃцрЃљрЃЊ рЃ«рЃЊрЃћрЃЉрЃљ, рЃарЃљрЃЊрЃњрЃљрЃю рЃЌрЃљрЃЋрЃўрЃЊрЃљрЃю рЃљрЃа рЃЋрЃЦрЃЏрЃюрЃўрЃЌ, рЃљрЃарЃљрЃЏрЃћрЃЊ рЃЋрЃЎрЃџрЃЮрЃюрЃљрЃЋрЃЌ рЃБрЃЎрЃЋрЃћ рЃљрЃарЃАрЃћрЃЉрЃБрЃџрЃА.
2. **рЃарЃћрЃАрЃБрЃџрЃарЃћрЃАрЃЉрЃўрЃА рЃЊрЃљрЃќрЃЮрЃњрЃЋрЃљ** - рЃЌрЃљрЃЋрЃўрЃЊрЃљрЃю рЃљрЃфрЃўрЃџрЃћрЃЉрЃА рЃЎрЃЮрЃЏрЃърЃџрЃћрЃЦрЃАрЃБрЃарЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃА рЃЌрЃљрЃЋрЃўрЃЊрЃљрЃю рЃерЃћрЃЦрЃЏрЃюрЃљрЃА, рЃарЃљрЃф рЃќрЃЮрЃњрЃљрЃЋрЃА рЃЏрЃћрЃ«рЃАрЃўрЃћрЃарЃћрЃЉрЃљрЃАрЃљ рЃЊрЃљ рЃърЃарЃЮрЃфрЃћрЃАрЃЮрЃарЃўрЃА рЃарЃћрЃАрЃБрЃарЃАрЃћрЃЉрЃА.
3. **рЃЏрЃЮрЃЊрЃБрЃџрЃљрЃарЃЮрЃЉрЃљ** - рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃА рЃбрЃўрЃърЃўрЃА рЃљрЃю рЃерЃћрЃЦрЃЏрЃюрЃўрЃА рЃърЃарЃЮрЃфрЃћрЃАрЃўрЃА рЃерЃћрЃфрЃЋрЃџрЃљ рЃерЃћрЃАрЃљрЃФрЃџрЃћрЃЉрЃћрЃџрЃўрЃљ рЃЎрЃџрЃЮрЃюрЃўрЃарЃћрЃЉрЃўрЃА рЃњрЃќрЃўрЃЌ/рЃЊрЃарЃЮрЃА.
4. **рЃцрЃљрЃарЃЌрЃЮ рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ** - рЃњрЃљрЃЏрЃЮрЃАрЃљрЃЊрЃћрЃњрЃўрЃљ рЃЌрЃљрЃЏрЃљрЃерЃћрЃЉрЃерЃў, рЃњрЃарЃљрЃцрЃўрЃЎрЃљрЃерЃў, рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃЌрЃљ рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃћрЃЉрЃерЃў, рЃАрЃљрЃЊрЃљрЃф рЃ«рЃерЃўрЃарЃљрЃЊ рЃАрЃљрЃГрЃўрЃарЃЮрЃљ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃЎрЃЮрЃърЃўрЃарЃћрЃЉрЃљ.

---

**­Ъџђ рЃЊрЃљрЃАрЃЎрЃЋрЃюрЃљ**

Prototype рЃърЃљрЃбрЃћрЃарЃюрЃў PHP-рЃерЃў рЃФрЃџрЃўрЃћрЃарЃў рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃљрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃњрЃљрЃюрЃАрЃљрЃЎрЃБрЃЌрЃарЃћрЃЉрЃўрЃЌ рЃњрЃљрЃЏрЃЮрЃАрЃљрЃЊрЃћрЃњрЃўрЃљ, рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃАрЃљрЃГрЃўрЃарЃЮрЃљ рЃарЃЌрЃБрЃџрЃў рЃљрЃю рЃФрЃЋрЃўрЃарЃљрЃЊрЃдрЃўрЃарЃћрЃЉрЃБрЃџрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃЎрЃџрЃЮрЃюрЃўрЃарЃћрЃЉрЃљ. Clone рЃцрЃБрЃюрЃЦрЃфрЃўрЃўрЃА рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃўрЃЌ рЃерЃћрЃАрЃљрЃФрЃџрЃћрЃЉрЃћрЃџрЃўрЃљ рЃЏрЃљрЃарЃбрЃўрЃЋрЃў рЃЊрЃљ рЃћрЃцрЃћрЃЦрЃбрЃБрЃарЃў рЃарЃћрЃљрЃџрЃўрЃќрЃљрЃфрЃўрЃљ, рЃарЃљрЃф рЃљрЃБрЃЏрЃ»рЃЮрЃЉрЃћрЃАрЃћрЃЉрЃА рЃЎрЃЮрЃЊрЃўрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљрЃАрЃљ рЃЊрЃљ рЃерЃћрЃАрЃарЃўрЃџрЃћрЃЉрЃўрЃА рЃАрЃўрЃЕрЃЦрЃљрЃарЃћрЃА.

Prototype рЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃўрЃЌ рЃерЃћрЃњрЃЋрЃўрЃФрЃџрЃўрЃљ рЃњрЃљрЃЋрЃљрЃЏрЃљрЃарЃбрЃўрЃЋрЃЮрЃЌ рЃЎрЃЮрЃЊрЃўрЃА рЃљрЃарЃЦрЃўрЃбрЃћрЃЦрЃбрЃБрЃарЃљ рЃЊрЃљ рЃБрЃЎрЃћрЃЌрЃћрЃАрЃљрЃЊ рЃЋрЃЏрЃљрЃарЃЌрЃЮрЃЌ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃў, рЃњрЃљрЃюрЃАрЃљрЃЎрЃБрЃЌрЃарЃћрЃЉрЃўрЃЌ рЃЏрЃљрЃерЃўрЃю, рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃЏрЃљрЃЌрЃў рЃерЃћрЃЦрЃЏрЃюрЃљ рЃЊрЃўрЃЊ рЃарЃћрЃАрЃБрЃарЃАрЃА рЃЏрЃЮрЃўрЃЌрЃ«рЃЮрЃЋрЃА.
