**Facade Pattern** рЃљрЃарЃўрЃА **рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃБрЃџрЃў рЃЊрЃўрЃќрЃљрЃўрЃю рЃърЃљрЃбрЃћрЃарЃюрЃўрЃА** рЃбрЃўрЃърЃў, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃњрЃЋрЃЌрЃљрЃЋрЃљрЃќрЃЮрЃЉрЃА рЃњрЃљрЃЏрЃљрЃарЃбрЃўрЃЋрЃћрЃЉрЃБрЃџ рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃА рЃЎрЃЮрЃЏрЃърЃџрЃћрЃЦрЃАрЃБрЃџрЃа рЃАрЃўрЃАрЃбрЃћрЃЏрЃћрЃЉрЃЌрЃљрЃю рЃБрЃарЃЌрЃўрЃћрЃарЃЌрЃЮрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА. рЃўрЃњрЃў рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћрЃЉрЃљ рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃАрЃљрЃГрЃўрЃарЃЮрЃљ рЃАрЃўрЃАрЃбрЃћрЃЏрЃўрЃА рЃерЃўрЃЊрЃљ рЃАрЃўрЃарЃЌрЃБрЃџрЃћрЃћрЃЉрЃўрЃА рЃЊрЃљрЃЏрЃљрЃџрЃЋрЃљ рЃЊрЃљ рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃџрЃўрЃАрЃЌрЃЋрЃўрЃА рЃЏрЃљрЃарЃбрЃўрЃЋрЃў рЃЊрЃљ рЃўрЃюрЃбрЃўрЃБрЃфрЃўрЃБрЃџрЃў API-рЃА рЃерЃћрЃЌрЃљрЃЋрЃљрЃќрЃћрЃЉрЃљ.

#### рЃарЃЮрЃЊрЃўрЃА рЃњрЃљрЃЏрЃЮрЃЋрЃўрЃДрЃћрЃюрЃЮрЃЌ Facade Pattern?
- рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃАрЃўрЃАрЃбрЃћрЃЏрЃљ рЃерЃћрЃЊрЃњрЃћрЃЉрЃљ рЃЉрЃћрЃЋрЃарЃў рЃЊрЃљрЃЏрЃЮрЃЎрЃўрЃЊрЃћрЃЉрЃБрЃџрЃћрЃЉрЃўрЃАрЃњрЃљрЃю рЃЊрЃљ рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃћрЃА рЃАрЃГрЃўрЃарЃЊрЃћрЃЉрЃљ рЃЏрЃљрЃарЃбрЃўрЃЋрЃў рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃў.
- рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃњрЃЋрЃАрЃБрЃарЃА рЃљрЃарЃАрЃћрЃЉрЃБрЃџрЃў рЃЎрЃЮрЃЊрЃўрЃА рЃЮрЃарЃњрЃљрЃюрЃўрЃќрЃћрЃЉрЃљ рЃЊрЃљ рЃАрЃўрЃЏрЃљрЃарЃбрЃўрЃЋрЃўрЃА рЃњрЃљрЃќрЃарЃЊрЃљ.
- рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃњрЃЋрЃАрЃБрЃарЃА рЃЏрЃарЃљрЃЋрЃљрЃџрЃерЃарЃўрЃљрЃюрЃў (Layered) рЃљрЃарЃЦрЃўрЃбрЃћрЃЦрЃбрЃБрЃарЃўрЃА рЃЊрЃљрЃюрЃћрЃарЃњрЃЋрЃљ.

---

#### Facade рЃърЃљрЃбрЃћрЃарЃюрЃўрЃА рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃљ

**рЃгрЃљрЃарЃЏрЃЮрЃЋрЃўрЃЊрЃњрЃўрЃюрЃЮрЃЌ рЃАрЃфрЃћрЃюрЃљрЃарЃў: рЃЋрЃўрЃЊрЃћрЃЮ рЃцрЃљрЃўрЃџрЃўрЃА рЃњрЃљрЃЊрЃљрЃДрЃЋрЃљрЃюрЃљ рЃАрЃ«рЃЋрЃљ рЃцрЃЮрЃарЃЏрЃљрЃбрЃерЃў**

##### 1. рЃарЃЌрЃБрЃџрЃў рЃЦрЃЋрЃћрЃАрЃўрЃАрЃбрЃћрЃЏрЃћрЃЉрЃў (рЃЊрЃљрЃЏрЃ«рЃЏрЃљрЃарЃћ рЃЎрЃџрЃљрЃАрЃў)

рЃАрЃўрЃАрЃбрЃћрЃЏрЃљрЃА рЃљрЃЦрЃЋрЃА рЃАрЃ«рЃЋрЃљрЃЊрЃљрЃАрЃ«рЃЋрЃљ рЃЎрЃџрЃљрЃАрЃћрЃЉрЃў, рЃарЃЮрЃЏрЃџрЃћрЃЉрЃўрЃф рЃАрЃљрЃГрЃўрЃарЃЮрЃљ рЃЋрЃўрЃЊрЃћрЃЮрЃА рЃЎрЃЮрЃюрЃЋрЃћрЃарЃбрЃљрЃфрЃўрЃўрЃАрЃЌрЃЋрЃўрЃА:

```php
<?php

class VideoFile {
    private string $filename;

    public function __construct(string $filename) {
        $this->filename = $filename;
    }

    public function getFilename(): string {
        return $this->filename;
    }
}

class CodecFactory {
    public static function getCodec(string $format) {
        if ($format === "mp4") {
            return new MP4Codec();
        } elseif ($format === "avi") {
            return new AVICodec();
        }
        throw new Exception("Unsupported format!");
    }
}

class MP4Codec {
    public function compress(VideoFile $file) {
        echo "Compressing {$file->getFilename()} to MP4 format.\n";
    }
}

class AVICodec {
    public function compress(VideoFile $file) {
        echo "Compressing {$file->getFilename()} to AVI format.\n";
    }
}

class AudioMixer {
    public function mix() {
        echo "Mixing audio...\n";
    }
}

class VideoConverter {
    public function convert(VideoFile $file, string $format) {
        echo "Converting {$file->getFilename()} to {$format}...\n";
        $codec = CodecFactory::getCodec($format);
        $codec->compress($file);
        (new AudioMixer())->mix();
        echo "Conversion complete!\n";
    }
}
```

рЃќрЃћрЃЏрЃЮрЃЌ рЃЏрЃЮрЃфрЃћрЃЏрЃБрЃџрЃў рЃЎрЃџрЃљрЃАрЃћрЃЉрЃў рЃБрЃарЃЌрЃўрЃћрЃарЃЌрЃЦрЃЏрЃћрЃЊрЃћрЃЉрЃћрЃю, рЃЏрЃљрЃњрЃарЃљрЃЏ рЃўрЃАрЃўрЃюрЃў рЃЊрЃљрЃЏрЃЮрЃБрЃЎрЃўрЃЊрЃћрЃЉрЃћрЃџрЃўрЃљ рЃЊрЃљ рЃАрЃљрЃЎрЃЏрЃљрЃЮрЃЊ рЃарЃЌрЃБрЃџрЃўрЃљ рЃЏрЃљрЃЌрЃў рЃБрЃерЃБрЃљрЃџрЃЮрЃЊ рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ.

---

##### 2. Facade рЃЎрЃџрЃљрЃАрЃў: рЃЋрЃўрЃЊрЃћрЃЮ рЃЎрЃЮрЃюрЃЋрЃћрЃарЃбрЃљрЃфрЃўрЃўрЃА рЃЏрЃљрЃарЃбрЃўрЃЋрЃў API

```php
<?php

class VideoConversionFacade {
    private VideoConverter $converter;

    public function __construct() {
        $this->converter = new VideoConverter();
    }

    public function convertVideo(string $filename, string $format) {
        $file = new VideoFile($filename);
        $this->converter->convert($file, $format);
    }
}
```

##### 3. Facade рЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ 

```php
<?php

$converter = new VideoConversionFacade();
$converter->convertVideo("sample_video.mov", "mp4");

```

#### рЃАрЃљрЃарЃњрЃћрЃЉрЃћрЃџрЃў Facade Pattern рЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃўрЃЌ:

РюЁ **рЃЏрЃљрЃарЃбрЃўрЃЋрЃў рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃў** - рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃћрЃџрЃА рЃЏрЃ«рЃЮрЃџрЃЮрЃЊ VideoConversionFacade рЃЎрЃџрЃљрЃАрЃЌрЃљрЃю рЃБрЃарЃЌрЃўрЃћрЃарЃЌрЃЮрЃЉрЃА.    
РюЁ **рЃерЃўрЃЊрЃљ рЃАрЃўрЃарЃЌрЃБрЃџрЃћрЃћрЃЉрЃўрЃА рЃЊрЃљрЃЏрЃљрЃџрЃЋрЃљ** - рЃарЃЌрЃБрЃџрЃў рЃЦрЃЋрЃћрЃАрЃўрЃАрЃбрЃћрЃЏрЃћрЃЉрЃў рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃџрЃўрЃАрЃЌрЃЋрЃўрЃА рЃБрЃ«рЃўрЃџрЃљрЃЋрЃўрЃљ.    
РюЁ **рЃЏрЃЮрЃЊрЃБрЃџрЃљрЃарЃЮрЃЉрЃљ** - рЃерЃћрЃАрЃљрЃФрЃџрЃћрЃЉрЃћрЃџрЃўрЃљ рЃЦрЃЋрЃћрЃАрЃўрЃАрЃбрЃћрЃЏрЃћрЃЉрЃўрЃА рЃерЃћрЃфрЃЋрЃџрЃљ Facade-рЃўрЃА API-рЃўрЃА рЃерЃћрЃфрЃЋрЃџрЃўрЃА рЃњрЃљрЃарЃћрЃерЃћ.    
РюЁ **рЃБрЃЎрЃћрЃЌрЃћрЃАрЃў рЃЎрЃЮрЃЊрЃўрЃА рЃЮрЃарЃњрЃљрЃюрЃўрЃќрЃћрЃЉрЃљ** - рЃБрЃцрЃарЃЮ рЃЏрЃљрЃарЃбрЃўрЃЋрЃў рЃЊрЃљ рЃњрЃљрЃАрЃљрЃњрЃћрЃЉрЃў рЃЎрЃЮрЃЊрЃў.

#### ­Ъџђ рЃЊрЃљрЃАрЃЎрЃЋрЃюрЃљ

**Facade Pattern** рЃФрЃљрЃџрЃўрЃљрЃю рЃњрЃљрЃЏрЃЮрЃАрЃљрЃЊрЃћрЃњрЃўрЃљ, рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃњрЃЋрЃАрЃБрЃарЃА рЃЎрЃЮрЃЏрЃърЃџрЃћрЃЦрЃАрЃБрЃарЃў рЃАрЃўрЃАрЃбрЃћрЃЏрЃўрЃА рЃњрЃљрЃЏрЃљрЃарЃбрЃўрЃЋрЃћрЃЉрЃљ рЃЊрЃљ рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃџрЃўрЃАрЃЌрЃЋрЃўрЃА рЃўрЃюрЃбрЃБрЃўрЃфрЃўрЃБрЃарЃў рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃўрЃА рЃерЃћрЃЌрЃљрЃЋрЃљрЃќрЃћрЃЉрЃљ. PHP рЃерЃў рЃЏрЃўрЃАрЃў рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ рЃњрЃљрЃюрЃАрЃљрЃЎрЃБрЃЌрЃарЃћрЃЉрЃўрЃЌ рЃАрЃљрЃАрЃљрЃарЃњрЃћрЃЉрЃџрЃЮрЃљрЃЊ, рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃЋрЃЏрЃБрЃерЃљрЃЮрЃЉрЃЌ рЃЏрЃарЃљрЃЋрЃљрЃџрЃерЃарЃўрЃљрЃю рЃљрЃарЃЦрЃўрЃбрЃћрЃЦрЃбрЃБрЃарЃљрЃќрЃћ, API-рЃћрЃЉрЃќрЃћ рЃљрЃю рЃарЃЌрЃБрЃџ рЃЉрЃўрЃќрЃюрЃћрЃА рЃџрЃЮрЃњрЃўрЃЎрЃљрЃќрЃћ.
