**Composite рЃърЃљрЃбрЃћрЃарЃюрЃў** рЃљрЃарЃўрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃБрЃџрЃў рЃЊрЃўрЃќрЃљрЃўрЃюрЃўрЃА рЃърЃљрЃбрЃћрЃарЃюрЃў,  рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃњрЃЋрЃљрЃФрЃџрЃћрЃЋрЃА рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃљрЃА рЃћрЃарЃЌрЃў рЃЊрЃљ рЃўрЃњрЃўрЃЋрЃћ рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃўрЃЌ рЃЋрЃЏрЃљрЃарЃЌрЃЮрЃЌ рЃарЃЮрЃњрЃЮрЃарЃф рЃўрЃюрЃЊрЃўрЃЋрЃўрЃЊрЃБрЃљрЃџрЃБрЃарЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃў, рЃўрЃАрЃћ рЃЏрЃљрЃЌрЃў рЃЎрЃЮрЃџрЃћрЃЦрЃфрЃўрЃћрЃЉрЃў. рЃћрЃА рЃърЃљрЃбрЃћрЃарЃюрЃў рЃњрЃљрЃюрЃАрЃљрЃЎрЃБрЃЌрЃарЃћрЃЉрЃўрЃЌ рЃњрЃљрЃЏрЃЮрЃАрЃљрЃЊрЃћрЃњрЃўрЃљ рЃўрЃћрЃарЃљрЃарЃЦрЃўрЃБрЃџрЃў рЃАрЃбрЃарЃўрЃЦрЃбрЃўрЃарЃћрЃЉрЃўрЃА, рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃљрЃЊ, рЃ«рЃћрЃћрЃЉрЃўрЃА (Tree Structures) рЃерЃћрЃАрЃљрЃЦрЃЏрЃюрЃџрЃћрЃљрЃЊ.

PHP-рЃерЃў **Composite Pattern** рЃњрЃЋрЃљрЃФрЃџрЃћрЃЋрЃА рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃљрЃА, рЃЊрЃљрЃЋрЃљрЃЏрЃБрЃерЃљрЃЋрЃЮрЃЌ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљ рЃўрЃАрЃћ, рЃЌрЃўрЃЌрЃЦрЃЮрЃА рЃўрЃАрЃўрЃюрЃў рЃћрЃарЃЌрЃўрЃљрЃюрЃў рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃўрЃљ. рЃЏрЃЮрЃЊрЃўрЃЌ, рЃњрЃљрЃюрЃЋрЃўрЃ«рЃўрЃџрЃЮрЃЌ рЃЏрЃўрЃАрЃў рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ рЃарЃћрЃљрЃџрЃБрЃа рЃърЃарЃЮрЃћрЃЦрЃбрЃерЃў.

---

#### рЃарЃЮрЃЊрЃўрЃА рЃњрЃљрЃЏрЃЮрЃЋрЃўрЃДрЃћрЃюрЃЮрЃЌ Composite Pattern?

- рЃарЃЮрЃфрЃљ рЃњрЃЋрЃАрЃБрЃарЃА рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљрЃА **рЃўрЃћрЃарЃљрЃарЃЦрЃўрЃБрЃџрЃў рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљ** рЃЏрЃўрЃЋрЃфрЃћрЃЌ (рЃЏрЃљрЃњ., рЃцрЃљрЃўрЃџрЃћрЃЉрЃўрЃА рЃАрЃўрЃАрЃбрЃћрЃЏрЃљ, рЃЏрЃћрЃюрЃўрЃБрЃћрЃЉрЃў, рЃњрЃарЃљрЃцрЃўрЃЎрЃБрЃџрЃў рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃћрЃЉрЃў).
- рЃарЃЮрЃфрЃљ рЃАрЃљрЃГрЃўрЃарЃЮрЃљ **рЃўрЃюрЃЊрЃўрЃЋрЃўрЃЊрЃБрЃљрЃџрЃБрЃарЃў рЃЊрЃљ рЃ»рЃњрЃБрЃцрЃБрЃарЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃћрЃарЃЌрЃўрЃљрЃюрЃљрЃЊ рЃЊрЃљрЃЏрЃБрЃерЃљрЃЋрЃћрЃЉрЃљ.**
- рЃарЃЮрЃфрЃљ рЃњрЃЋрЃАрЃБрЃарЃА **рЃЏрЃЮрЃЊрЃБрЃџрЃљрЃарЃЮрЃЉрЃљ рЃЊрЃљ рЃЏрЃЮрЃЦрЃюрЃўрЃџрЃЮрЃЉрЃљ**, рЃарЃљрЃЌрЃљ рЃАрЃљрЃГрЃўрЃарЃЮрЃћрЃЉрЃўрЃАрЃљрЃЏрЃћрЃЉрЃа рЃЊрЃљрЃЋрЃљрЃЏрЃљрЃбрЃЮрЃЌ рЃљрЃ«рЃљрЃџрЃў рЃЎрЃЮрЃЏрЃърЃЮрЃюрЃћрЃюрЃбрЃћрЃЉрЃў.
---

#### рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў, рЃцрЃљрЃўрЃџрЃћрЃЉрЃўрЃА рЃАрЃўрЃАрЃбрЃћрЃЏрЃљ

рЃњрЃљрЃЋрЃљрЃЎрЃћрЃЌрЃЮрЃЌ рЃцрЃљрЃўрЃџрЃћрЃЉрЃўрЃАрЃљ рЃЊрЃљ рЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃўрЃћрЃЉрЃўрЃА рЃАрЃбрЃарЃБрЃфрЃбрЃБрЃарЃљ **Composite Pattern** рЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃўрЃЌ.

##### 1. рЃЋрЃЦрЃЏрЃюрЃўрЃЌ рЃЎрЃЮрЃЏрЃърЃЮрЃюрЃћрЃюрЃбрЃА (рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃА)
```php
<?php

interface FileSystemComponent {
    public function show();
}
```

рЃћрЃА рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃў рЃДрЃЋрЃћрЃџрЃљ рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃА (рЃарЃЮрЃњрЃЮрЃарЃф рЃцрЃљрЃўрЃџрЃћрЃЉрЃА, рЃўрЃАрЃћ рЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃўрЃћрЃЉрЃА) рЃљрЃЋрЃљрЃџрЃЊрЃћрЃЉрЃБрЃџрЃћрЃЉрЃА, рЃарЃЮрЃЏ рЃ░рЃЦрЃЮрЃюрЃЊрЃћрЃЌ **show()** рЃЏрЃћрЃЌрЃЮрЃЊрЃў.

##### 2. рЃЋрЃЦрЃЏрЃюрЃўрЃЌ рЃцрЃљрЃўрЃџрЃўрЃА рЃЎрЃџрЃљрЃАрЃА

```php
class File implements FileSystemComponent {
    private string $name;

    public function __construct(string $name) {
        $this->name = $name;
    }

    public function show() {
        echo "File: " . $this->name . PHP_EOL;
    }
}
```

##### 3. рЃЋрЃЦрЃЏрЃюрЃўрЃЌ рЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃўрЃўрЃА рЃЎрЃџрЃљрЃАрЃА

```php
class Directory implements FileSystemComponent {
    private string $name;
    private array $children = [];

    public function __construct(string $name) {
        $this->name = $name;
    }

    public function add(FileSystemComponent $component) {
        $this->children[] = $component;
    }

    public function show() {
        echo "Directory: " . $this->name . PHP_EOL;
        foreach ($this->children as $child) {
            echo "  ";
            $child->show();
        }
    }
}
```

**Directory** рЃЎрЃџрЃљрЃАрЃў **рЃЎрЃЮрЃЏрЃърЃЮрЃќрЃўрЃбрЃўрЃљ** рЃЊрЃљ рЃерЃћрЃўрЃФрЃџрЃћрЃЉрЃљ рЃерЃћрЃўрЃфрЃљрЃЋрЃЊрЃћрЃА рЃарЃЮрЃњрЃЮрЃарЃф рЃцрЃљрЃўрЃџрЃћрЃЉрЃА, рЃўрЃАрЃћ рЃАрЃ«рЃЋрЃљ рЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃўрЃћрЃЉрЃА.

##### 4. рЃЋрЃўрЃДрЃћрЃюрЃћрЃЉрЃЌ рЃЕрЃЋрЃћрЃюрЃА рЃърЃљрЃбрЃћрЃарЃюрЃА

```php
// рЃўрЃюрЃЊрЃўрЃЋрЃўрЃЊрЃБрЃљрЃџрЃБрЃарЃў рЃцрЃљрЃўрЃџрЃћрЃЉрЃў
$file1 = new File("document.txt");
$file2 = new File("photo.jpg");

// рЃАрЃљрЃЦрЃљрЃдрЃљрЃџрЃЊрЃћ (рЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃўрЃљ)
$folder1 = new Directory("My Documents");
$folder1->add($file1);
$folder1->add($file2);

// рЃЏрЃћрЃЮрЃарЃћ рЃАрЃљрЃЦрЃљрЃдрЃљрЃџрЃЊрЃћ (рЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃўрЃљ)
$folder2 = new Directory("Projects");
$file3 = new File("project_code.php");
$folder2->add($file3);

// рЃЏрЃЌрЃљрЃЋрЃљрЃарЃў рЃАрЃљрЃЦрЃљрЃдрЃљрЃџрЃЊрЃћ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃерЃћрЃўрЃфрЃљрЃЋрЃА рЃЮрЃа рЃАрЃљрЃЦрЃљрЃдрЃљрЃџрЃЊрЃћрЃА
$root = new Directory("Root");
$root->add($folder1);
$root->add($folder2);

// рЃЋрЃљрЃЕрЃЋрЃћрЃюрЃћрЃЉрЃЌ рЃЏрЃЌрЃџрЃўрЃљрЃю рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљрЃА
$root->show();
```

##### 5. рЃерЃћрЃЊрЃћрЃњрЃў

рЃљрЃЏ рЃЎрЃЮрЃЊрЃўрЃА рЃњрЃљрЃерЃЋрЃћрЃЉрЃўрЃАрЃљрЃА рЃЏрЃўрЃЋрЃўрЃдрЃћрЃЉрЃЌ рЃерЃћрЃЏрЃЊрЃћрЃњ рЃерЃћрЃЊрЃћрЃњрЃА:
```php
Directory: Root
  Directory: My Documents
    File: document.txt
    File: photo.jpg
  Directory: Projects
    File: project_code.php
```

рЃарЃЮрЃњрЃЮрЃарЃф рЃЋрЃ«рЃћрЃЊрЃљрЃЋрЃЌ, **Directory** рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃА рЃерЃћрЃБрЃФрЃџрЃўрЃљ рЃерЃћрЃўрЃфрЃљрЃЋрЃЊрЃћрЃА рЃарЃЮрЃњрЃЮрЃарЃф **File** - рЃћрЃЉрЃА, рЃљрЃАрЃћ рЃАрЃ«рЃЋрЃљ **Directory** - рЃћрЃЉрЃА, рЃЊрЃљ рЃДрЃЋрЃћрЃџрЃљ рЃЏрЃљрЃЌрЃњрЃљрЃюрЃА рЃћрЃюрЃЌрЃюрЃљрЃўрЃарЃљрЃЊ рЃЋрЃљрЃЏрЃБрЃерЃљрЃЋрЃћрЃЉрЃЌ **show()** рЃЏрЃћрЃЌрЃЮрЃЊрЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃўрЃЌ.

#### ­Ъџђ рЃЊрЃљрЃАрЃЎрЃЋрЃюрЃљ


**Composite Pattern** PHP-рЃерЃў рЃФрЃљрЃџрЃўрЃљрЃю рЃЏрЃЮрЃАрЃљрЃ«рЃћрЃарЃ«рЃћрЃЉрЃћрЃџрЃўрЃљ рЃўрЃћрЃарЃљрЃарЃЦрЃўрЃБрЃџрЃў рЃАрЃбрЃарЃБрЃбрЃБрЃарЃћрЃЉрЃўрЃА рЃерЃћрЃАрЃљрЃЦрЃЏрЃюрЃћрЃџрЃљрЃЊ, рЃарЃљрЃф рЃњрЃЋрЃљрЃФрЃџрЃћрЃЋрЃА:
- **рЃЏрЃЮрЃЊрЃБрЃџрЃљрЃарЃЮрЃЉрЃљрЃА** - рЃерЃћрЃњрЃЋрЃўрЃФрЃџрЃўрЃљ рЃЏрЃљрЃарЃбрЃўрЃЋрЃљрЃЊ рЃЊрЃљрЃЋрЃљрЃЏрЃљрЃбрЃЮрЃЌ рЃљрЃю рЃгрЃљрЃЋрЃерЃљрЃџрЃЮрЃЌ рЃЎрЃЮрЃЏрЃърЃЮрЃюрЃћрЃюрЃбрЃћрЃЉрЃў.
- **рЃЏрЃЮрЃЦрЃюрЃўрЃџрЃЮрЃЉрЃљрЃА** - рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃћрЃА рЃљрЃа рЃАрЃГрЃўрЃарЃЊрЃћрЃЉрЃљ рЃўрЃфрЃЮрЃЊрЃћрЃА, рЃўрЃюрЃЊрЃўрЃЋрЃўрЃЊрЃБрЃљрЃџрЃўрЃа рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃЌрЃљрЃю рЃљрЃЦрЃЋрЃЌ рЃАрЃљрЃЦрЃЏрЃћ рЃЌрЃБ рЃЎрЃЮрЃЏрЃърЃЮрЃќрЃўрЃбрЃЌрЃљрЃю.
- **рЃњрЃљрЃЊрЃљрЃЏрЃБрЃерЃљрЃЋрЃћрЃЉрЃљрЃЊрЃЮрЃЉрЃљрЃА** - рЃЎрЃЮрЃЊрЃўрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљ рЃўрЃерЃџрЃћрЃЉрЃљ рЃЊрЃљ рЃЏрЃўрЃАрЃў рЃЏрЃљрЃарЃЌрЃЋрЃљ рЃЏрЃљрЃарЃбрЃўрЃЋрЃўрЃљ.

рЃћрЃА рЃърЃљрЃбрЃћрЃарЃюрЃў рЃљрЃЦрЃбрЃўрЃБрЃарЃљрЃЊ рЃЏрЃњрЃљрЃЏрЃЮрЃўрЃДрЃДрЃћрЃюрЃћрЃЉрЃљ рЃўрЃАрЃћрЃЌ рЃърЃарЃЮрЃћрЃЦрЃбрЃћрЃЉрЃерЃў, рЃарЃЮрЃњрЃЮрЃарЃўрЃфрЃљрЃљ:
- рЃцрЃљрЃўрЃџрЃћрЃЉрЃўрЃА рЃЏрЃљрЃарЃЌрЃЋрЃўрЃА рЃАрЃўрЃАрЃбрЃћрЃЏрЃћрЃЉрЃў.
- рЃЏрЃћрЃюрЃўрЃБрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃћрЃЉрЃў.
- рЃњрЃарЃљрЃцрЃўрЃЎрЃБрЃџрЃў рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃћрЃЉрЃўрЃА рЃЏрЃљрЃарЃЌрЃЋрЃљ.
- рЃЮрЃарЃњрЃљрЃюрЃўрЃќрЃљрЃфрЃўрЃБрЃџрЃў рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃћрЃЉрЃў.
