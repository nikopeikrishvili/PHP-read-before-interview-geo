
**Bridge (рЃ«рЃўрЃЊрЃў) рЃърЃљрЃбрЃћрЃарЃюрЃў** рЃљрЃарЃўрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃБрЃџрЃў рЃЊрЃўрЃќрЃљрЃўрЃю рЃърЃљрЃбрЃћрЃарЃюрЃў, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃЏрЃўрЃќрЃюрЃљрЃЊ рЃўрЃАрЃљрЃ«рЃљрЃЋрЃА рЃљрЃЉрЃАрЃбрЃарЃљрЃЦрЃфрЃўрЃўрЃА рЃЊрЃљ рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃўрЃА рЃњрЃљрЃюрЃфрЃљрЃџрЃЎрЃћрЃЋрЃћрЃЉрЃљрЃА, рЃарЃљрЃдрЃљ рЃЏрЃљрЃЌрЃў рЃЊрЃљрЃЏрЃЮрЃБрЃЎрЃўрЃЊрЃћрЃЉрЃћрЃџрЃў рЃњрЃљрЃюрЃЋрЃўрЃЌрЃљрЃарЃћрЃЉрЃљ рЃерЃћрЃАрЃљрЃФрЃџрЃћрЃЉрЃћрЃџрЃў рЃўрЃДрЃЮрЃА. рЃћрЃА рЃърЃљрЃбрЃћрЃарЃюрЃў рЃњрЃљрЃюрЃАрЃљрЃЎрЃБрЃЌрЃарЃћрЃћрЃЉрЃўрЃЌ рЃАрЃљрЃАрЃљрЃарЃњрЃћрЃЉрЃџрЃЮрЃљ, рЃарЃЮрЃфрЃљ рЃњрЃЋрЃўрЃюрЃЊрЃљ рЃЌрЃљрЃЋрЃўрЃЊрЃљрЃю рЃљрЃЋрЃўрЃфрЃўрЃџрЃЮрЃЌ рЃЎрЃџрЃљрЃАрЃћрЃЉрЃўрЃА рЃњрЃљрЃЊрЃљрЃГрЃљрЃарЃЉрЃћрЃЉрЃБрЃџрЃў рЃЏрЃћрЃЏрЃЎрЃЋрЃўрЃЊрЃарЃћрЃЮрЃЉрЃўрЃЌрЃЮрЃЉрЃљ рЃЊрЃљ рЃњрЃљрЃЋрЃќрЃљрЃарЃЊрЃЮрЃЌ рЃЎрЃЮрЃЊрЃўрЃА рЃЏрЃЮрЃЦрЃюрЃўрЃџрЃЮрЃЉрЃљ.

PHP-рЃерЃў **Bridge** рЃърЃљрЃбрЃћрЃарЃюрЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ рЃ«рЃћрЃџрЃА рЃБрЃгрЃДрЃЮрЃЉрЃА рЃАрЃБрЃцрЃЌрЃљ рЃљрЃарЃЦрЃўрЃбрЃћрЃЦрЃбрЃБрЃарЃљрЃА, рЃАрЃљрЃЊрЃљрЃф рЃерЃћрЃАрЃљрЃФрЃџрЃћрЃЉрЃћрЃџрЃў рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃўрЃА рЃЏрЃљрЃарЃбрЃўрЃЋрЃљрЃЊ рЃерЃћрЃ«рЃЋрЃџрЃљ рЃЎрЃЮрЃЊрЃўрЃА рЃ«рЃћрЃџрЃљрЃ«рЃљрЃџрЃў рЃњрЃљрЃЊрЃљрЃгрЃћрЃарЃўрЃА рЃњрЃљрЃарЃћрЃерЃћ.


#### рЃарЃЮрЃЊрЃўрЃА рЃњрЃљрЃЏрЃЮрЃЋрЃўрЃДрЃћрЃюрЃЮрЃЌ Bridge?
- **рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃњрЃЋрЃўрЃюрЃЊрЃљ рЃЊрЃљрЃЋрЃљрЃфрЃљрЃџрЃЎрЃћрЃЋрЃЮрЃЌ рЃљрЃЉрЃАрЃбрЃарЃљрЃЦрЃфрЃўрЃљ рЃЊрЃљ рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃљ** - рЃћрЃА рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃљрЃА рЃњрЃЋрЃљрЃФрЃџрЃћрЃЋрЃА, рЃарЃЮрЃЏ рЃЮрЃарЃўрЃЋрЃћ рЃЏрЃљрЃЌрЃњрЃљрЃюрЃў рЃЊрЃљрЃЏрЃЮрЃБрЃЎрЃўрЃЊрЃћрЃЉрЃџрЃљрЃЊ рЃерЃћрЃЋрЃфрЃЋрЃљрЃџрЃЮрЃЌ.
- **рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃЎрЃџрЃљрЃАрЃћрЃЉрЃўрЃА рЃЊрЃўрЃЊрЃў рЃЏрЃћрЃЏрЃЎрЃЋрЃўрЃЊрЃарЃћрЃЮрЃЉрЃўрЃЌрЃў рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљ рЃљрЃарЃЌрЃБрЃџрЃћрЃЉрЃА рЃЎрЃЮрЃЊрЃўрЃА рЃЏрЃљрЃарЃЌрЃЋрЃљрЃА** - Bridge рЃ«рЃћрЃџрЃА рЃБрЃерЃџрЃўрЃА рЃњрЃљрЃЊрЃљрЃГрЃљрЃарЃЉрЃћрЃЉрЃБрЃџрЃў рЃЏрЃћрЃЏрЃЎрЃЋрЃўрЃЊрЃарЃћрЃЮрЃЉрЃўрЃА рЃърЃарЃЮрЃЉрЃџрЃћрЃЏрЃћрЃЉрЃА.
- **рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃАрЃљрЃГрЃўрЃарЃЮрЃљ рЃАрЃ«рЃЋрЃљрЃЊрЃљрЃАрЃ«рЃЋрЃљ рЃърЃџрЃљрЃбрЃцрЃЮрЃарЃЏрЃљрЃќрЃћ рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃўрЃА рЃБрЃќрЃарЃБрЃюрЃЋрЃћрЃџрЃДрЃЮрЃцрЃљ** - рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃљрЃЊ, рЃћрЃарЃЌрЃў рЃЊрЃљ рЃўрЃњрЃўрЃЋрЃћ рЃљрЃЉрЃАрЃбрЃарЃљрЃЦрЃфрЃўрЃљ рЃерЃћрЃўрЃФрЃџрЃћрЃЉрЃљ рЃўрЃЏрЃБрЃерЃљрЃЮрЃА рЃАрЃ«рЃЋрЃљрЃЊрЃљрЃАрЃ«рЃЋрЃљ рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃЌрЃљ рЃЉрЃљрЃќрЃћрЃЉрЃЌрЃљрЃю.

#### Bridge рЃърЃљрЃбрЃћрЃарЃюрЃўрЃА рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃљ

##### 1. рЃљрЃЉрЃАрЃбрЃарЃљрЃЦрЃфрЃўрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ

рЃћрЃА рЃљрЃарЃўрЃА рЃЎрЃџрЃљрЃАрЃў, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃўрЃА рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃА рЃерЃћрЃўрЃфрЃљрЃЋрЃА рЃЊрЃљ рЃЏрЃљрЃАрЃќрЃћ рЃЮрЃърЃћрЃарЃљрЃфрЃўрЃћрЃЉрЃА рЃљрЃАрЃарЃБрЃџрЃћрЃЉрЃА.
```php
<?php

// рЃљрЃЉрЃАрЃбрЃарЃљрЃЦрЃбрЃБрЃџрЃў рЃЎрЃџрЃљрЃАрЃў (Abstraction)
abstract class Device {
    protected $implementation;

    public function __construct(DeviceImplementation $implementation) {
        $this->implementation = $implementation;
    }

    abstract public function operate();
}
```

##### 2. рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃўрЃА рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃў
рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃўрЃА рЃњрЃљрЃюрЃАрЃљрЃќрЃдрЃЋрЃарЃљ рЃ«рЃЊрЃћрЃЉрЃљ рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃўрЃА рЃЏрЃћрЃерЃЋрЃћрЃЮрЃЉрЃўрЃЌ.
```php
// рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃўрЃА рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃў (Implementation)
interface DeviceImplementation {
    public function turnOn();
    public function turnOff();
}
```

##### 3. рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃўрЃА рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџрЃў рЃЎрЃџрЃљрЃАрЃћрЃЉрЃў

рЃћрЃА рЃЎрЃџрЃљрЃАрЃћрЃЉрЃў рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃўрЃА рЃЊрЃћрЃбрЃљрЃџрЃћрЃЉрЃА рЃњрЃљрЃюрЃАрЃљрЃќрЃдрЃЋрЃарЃљрЃЋрЃћрЃю.
```php
// рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџрЃў рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃљ (Concrete Implementation)
class TV implements DeviceImplementation {
    public function turnOn() {
        echo "TV turned on\n";
    }

    public function turnOff() {
        echo "TV turned off\n";
    }
}

class Radio implements DeviceImplementation {
    public function turnOn() {
        echo "Radio turned on\n";
    }

    public function turnOff() {
        echo "Radio turned off\n";
    }
}
```

##### 4. рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџрЃў рЃљрЃЉрЃАрЃбрЃарЃљрЃЦрЃфрЃўрЃўрЃА рЃЎрЃџрЃљрЃАрЃћрЃЉрЃў

рЃљрЃЏ рЃЎрЃџрЃљрЃАрЃћрЃЉрЃерЃў рЃњрЃљрЃюрЃАрЃљрЃќрЃдрЃЋрЃарЃБрЃџрЃўрЃљ рЃљрЃЉрЃАрЃбрЃарЃљрЃЦрЃфрЃўрЃўрЃА рЃАрЃърЃћрЃфрЃўрЃцрЃўрЃЎрЃБрЃарЃў рЃЦрЃфрЃћрЃЋрЃљ.
```php
// рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџрЃў рЃљрЃЉрЃАрЃбрЃарЃљрЃЦрЃфрЃўрЃљ (Refined Abstraction)
class RemoteControl extends Device {
    public function operate() {
        echo "Using remote control:\n";
        $this->implementation->turnOn();
        $this->implementation->turnOff();
    }
}
```

##### 5. рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ

рЃЋрЃюрЃљрЃ«рЃЮрЃЌ, рЃарЃЮрЃњрЃЮрЃа рЃерЃћрЃњрЃЋрЃўрЃФрЃџрЃўрЃљ рЃљрЃЏ рЃЎрЃџрЃљрЃАрЃћрЃЉрЃў рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ.
```php
$tv = new TV();
$radio = new Radio();

$remoteForTV = new RemoteControl($tv);
$remoteForRadio = new RemoteControl($radio);

$remoteForTV->operate();
$remoteForRadio->operate();
```

##### Bridge рЃърЃљрЃбрЃћрЃарЃюрЃўрЃА рЃБрЃърЃўрЃарЃљрЃбрЃћрЃАрЃЮрЃЉрЃћрЃЉрЃў
- **рЃЏрЃЮрЃ«рЃАрЃюрЃўрЃџрЃў рЃЊрЃљрЃЏрЃЮрЃЎрЃўрЃЊрЃћрЃЉрЃБрЃџрЃћрЃЉрЃљ рЃљрЃЉрЃАрЃбрЃарЃљрЃЦрЃфрЃўрЃљрЃАрЃљрЃА рЃЊрЃљ рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃљрЃА рЃерЃЮрЃарЃўрЃА** - рЃерЃћрЃњрЃЋрЃўрЃФрЃџрЃўрЃљ рЃЊрЃљрЃЏрЃЮрЃБрЃЎрЃўрЃЊрЃћрЃЉрЃџрЃљрЃЊ рЃњрЃљрЃюрЃЋрЃљрЃЋрЃўрЃЌрЃљрЃарЃЮрЃЌ рЃЮрЃарЃўрЃЋрЃћ рЃЏрЃ«рЃљрЃарЃћ.
- **рЃЏрЃарЃљрЃЋрЃљрЃџрЃ»рЃћрЃарЃљрЃЊрЃў рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃўрЃА рЃерЃћрЃАрЃљрЃФрЃџрЃћрЃЉрЃџрЃЮрЃЉрЃљ** - рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃћрЃЉрЃА рЃАрЃ«рЃЋрЃљрЃЊрЃљрЃАрЃ«рЃЋрЃљ рЃљрЃЉрЃАрЃбрЃарЃљрЃЦрЃфрЃўрЃћрЃЉрЃЌрЃљрЃю рЃЏрЃљрЃарЃбрЃўрЃЋрЃљрЃЊ рЃЋрЃўрЃДрЃћрЃюрЃћрЃЉрЃЌ.
- **рЃЏрЃћрЃЏрЃЎрЃЋрЃўрЃЊрЃарЃћрЃЮрЃЉрЃўрЃА рЃАрЃўрЃдрЃарЃЏрЃўрЃА рЃерЃћрЃЏрЃфрЃўрЃарЃћрЃЉрЃљ** - рЃюрЃљрЃфрЃЋрЃџрЃљрЃЊ рЃўрЃЏрЃўрЃАрЃљ, рЃарЃЮрЃЏ рЃЉрЃћрЃЋрЃарЃў рЃЎрЃџрЃљрЃАрЃўрЃАрЃњрЃљрЃю рЃерЃћрЃЏрЃЊрЃњрЃљрЃарЃў рЃЏрЃћрЃЏрЃЎрЃЋрЃўрЃЊрЃарЃћрЃЮрЃЉрЃўрЃЌрЃў рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљ рЃерЃћрЃЋрЃЦрЃЏрЃюрЃљрЃЌ, рЃЋрЃўрЃДрЃћрЃюрЃћрЃЉрЃЌ рЃЎ0рЃЮрЃЏрЃърЃЮрЃќрЃўрЃфрЃўрЃљрЃА.

**Bridge рЃърЃљрЃбрЃћрЃарЃюрЃў** PHP - рЃерЃў рЃАрЃљрЃАрЃљрЃарЃњрЃћрЃЉрЃџрЃЮрЃљ, рЃарЃЮрЃфрЃљ рЃњрЃЋрЃАрЃБрЃарЃА рЃЏрЃћрЃЏрЃЎрЃЋрЃўрЃЊрЃарЃћрЃЮрЃЉрЃўрЃЌрЃЮрЃЉрЃўрЃА рЃАрЃўрЃарЃЌрЃБрЃџрЃћрЃћрЃЉрЃўрЃА рЃЌрЃљрЃЋрЃўрЃЊрЃљрЃю рЃљрЃфрЃўрЃџрЃћрЃЉрЃљ рЃЊрЃљ рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃўрЃА рЃЏрЃЮрЃЦрЃюрЃўрЃџрЃЮрЃЉрЃљ. рЃЏрЃўрЃАрЃў рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ рЃњрЃљрЃюрЃАрЃљрЃЎрЃБрЃЌрЃарЃћрЃЉрЃўрЃЌ рЃћрЃцрЃћрЃЦрЃбрЃБрЃарЃўрЃљ, рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃЎрЃЮрЃЊрЃўрЃА рЃњрЃљрЃцрЃљрЃарЃЌрЃЮрЃћрЃЉрЃљрЃА рЃЋрЃњрЃћрЃњрЃЏрЃљрЃЋрЃЌ рЃАрЃ«рЃЋрЃљрЃЊрЃљрЃАрЃ«рЃЋрЃљ рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃўрЃА рЃЏрЃ«рЃљрЃарЃЊрЃљрЃГрЃћрЃарЃўрЃЌ.
