
**Template Method** - рЃћрЃарЃЌ-рЃћрЃарЃЌрЃў рЃЏрЃюрЃўрЃерЃюрЃЋрЃюрЃћрЃџрЃЮрЃЋрЃљрЃюрЃў рЃЦрЃфрЃћрЃЋрЃўрЃЌрЃў рЃбрЃўрЃърЃўрЃА рЃЊрЃўрЃќрЃљрЃўрЃюрЃўрЃА рЃърЃљрЃбрЃћрЃарЃюрЃўрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћрЃЉрЃљ рЃљрЃџрЃњрЃЮрЃарЃўрЃЌрЃЏрЃўрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃўрЃА рЃњрЃљрЃюрЃАрЃљрЃќрЃдрЃЋрЃарЃљрЃА рЃЊрЃљ рЃЏрЃўрЃА рЃюрЃљрЃгрЃўрЃџрЃЮрЃЉрЃарЃўрЃЋ рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃљрЃА.

Template Method рЃњрЃљрЃюрЃАрЃљрЃЎрЃБрЃЌрЃарЃћрЃЉрЃўрЃЌ рЃАрЃљрЃАрЃљрЃарЃњрЃћрЃЉрЃџрЃЮрЃљрЃЊ, рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃњрЃЋрЃљрЃЦрЃЋрЃА рЃАрЃљрЃћрЃарЃЌрЃЮ рЃърЃарЃЮрЃфрЃћрЃАрЃўрЃА рЃарЃљрЃЏрЃЊрЃћрЃюрЃўрЃЏрЃљ рЃЋрЃљрЃарЃўрЃљрЃюрЃбрЃў, рЃарЃЮрЃЏрЃћрЃџрЃЌрЃљ рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљ рЃћрЃарЃЌрЃюрЃљрЃўрЃарЃўрЃљ, рЃЏрЃљрЃњрЃарЃљрЃЏ рЃќрЃЮрЃњрЃўрЃћрЃЌрЃў рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџрЃў рЃюрЃљрЃЉрЃўрЃ»рЃў рЃњрЃљрЃюрЃАрЃќрЃЋрЃљрЃЋрЃЊрЃћрЃЉрЃљ.

---

#### Template Method - рЃўрЃА рЃърЃарЃўрЃюрЃфрЃўрЃърЃў

Template Method рЃърЃљрЃбрЃћрЃарЃюрЃў рЃњрЃБрЃџрЃўрЃАрЃ«рЃЏрЃЮрЃЉрЃА, рЃарЃЮрЃЏ:

1. рЃњрЃљрЃюрЃЋрЃАрЃљрЃќрЃдрЃЋрЃарЃЮрЃЌ рЃљрЃџрЃњрЃЮрЃарЃўрЃЌрЃЏрЃўрЃА рЃФрЃўрЃарЃўрЃЌрЃљрЃЊрЃў рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљ рЃљрЃЉрЃАрЃбрЃарЃљрЃЦрЃбрЃБрЃџ рЃЎрЃџрЃљрЃАрЃерЃў.
2. рЃљрЃџрЃњрЃЮрЃарЃўрЃЌрЃЏрЃўрЃА рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџрЃў рЃюрЃљрЃгрЃўрЃџрЃў рЃГрЃбрЃљрЃЏрЃЮрЃЏрЃљрЃЋрЃљрЃџ рЃЎрЃџрЃљрЃАрЃћрЃЉрЃерЃў рЃњрЃљрЃЏрЃЮрЃЋрЃДрЃЮрЃЌ, рЃарЃљрЃЌрЃљ рЃЏрЃљрЃЌрЃў рЃЦрЃфрЃћрЃЋрЃўрЃА рЃЊрЃљрЃЎрЃЮрЃюрЃцрЃўрЃњрЃБрЃарЃўрЃарЃћрЃЉрЃљ рЃўрЃДрЃЮрЃА рЃерЃћрЃАрЃљрЃФрЃџрЃћрЃЉрЃћрЃџрЃў.
3. рЃќрЃЮрЃњрЃљрЃЊрЃў рЃљрЃџрЃњрЃЮрЃарЃўрЃЌрЃЏрЃў рЃљрЃа рЃерЃћрЃўрЃфрЃЋрЃџрЃћрЃЉрЃљ, рЃ«рЃЮрЃџрЃЮ рЃерЃЋрЃўрЃџрЃЮрЃЉрЃўрЃџ рЃЎрЃџрЃљрЃАрЃћрЃЉрЃА рЃерЃћрЃБрЃФрЃџрЃўрЃљрЃЌ рЃЌрЃљрЃЋрЃўрЃљрЃюрЃЌрЃў рЃАрЃљрЃГрЃўрЃарЃЮрЃћрЃЉрЃћрЃЉрЃўрЃА рЃЏрЃўрЃ«рЃћрЃЊрЃЋрЃўрЃЌ рЃерЃћрЃфрЃЋрЃљрЃџрЃЮрЃю рЃљрЃю рЃЊрЃљрЃљрЃЏрЃљрЃбрЃЮрЃю рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџрЃў рЃЦрЃЏрЃћрЃЊрЃћрЃЉрЃћрЃЉрЃў.

---
#### Template Method - рЃўрЃА рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃљ

рЃЊрЃљрЃЋрЃБрЃерЃЋрЃљрЃЌ, рЃарЃЮрЃЏ рЃЋрЃЦрЃЏрЃюрЃўрЃЌ **рЃарЃћрЃърЃЮрЃарЃбрЃўрЃюрЃњрЃўрЃА рЃАрЃўрЃАрЃбрЃћрЃЏрЃљрЃА**, рЃАрЃљрЃЊрЃљрЃф рЃАрЃ«рЃЋрЃљрЃЊрЃљрЃАрЃ«рЃЋрЃљ рЃцрЃЮрЃарЃЏрЃљрЃбрЃўрЃА рЃарЃћрЃърЃЮрЃарЃбрЃўрЃА (CSV, HTML) рЃњрЃћрЃюрЃћрЃарЃљрЃфрЃўрЃљ рЃБрЃюрЃЊрЃљ рЃерЃћрЃЋрЃФрЃџрЃЮрЃЌ.

##### 1. рЃљрЃЉрЃАрЃбрЃарЃљрЃЦрЃбрЃБрЃџрЃў рЃЎрЃџрЃљрЃАрЃў - ReportGenerator
```php
abstract class ReportGenerator {
    // Template Method РђЊ рЃњрЃљрЃюрЃАрЃљрЃќрЃдрЃЋрЃарЃљрЃЋрЃА рЃљрЃюрЃњрЃљрЃарЃўрЃерЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃўрЃА рЃќрЃЮрЃњрЃљрЃЊ рЃърЃарЃЮрЃфрЃћрЃАрЃА
    public function generateReport() {
        $data = $this->fetchData();
        $formattedData = $this->formatData($data);
        $this->exportReport($formattedData);
    }

    // рЃљрЃЉрЃАрЃбрЃарЃљрЃЦрЃбрЃБрЃџрЃў рЃЏрЃћрЃЌрЃЮрЃЊрЃћрЃЉрЃў, рЃарЃЮрЃЏрЃџрЃћрЃЉрЃўрЃф рЃерЃЋрЃўрЃџрЃЮрЃЉрЃўрЃџ рЃЎрЃџрЃљрЃАрЃћрЃЉрЃерЃў рЃБрЃюрЃЊрЃљ рЃњрЃљрЃюрЃЋрЃАрЃљрЃќрЃдрЃЋрЃарЃЮрЃЌ
    abstract protected function fetchData();
    abstract protected function formatData($data);
    abstract protected function exportReport($formattedData);
}
```

рЃљрЃЦ **generateReport()** рЃљрЃарЃўрЃА **Template Method**, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃњрЃљрЃюрЃАрЃљрЃќрЃдрЃЋрЃарЃљрЃЋрЃА рЃарЃћрЃърЃЮрЃарЃбрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃўрЃА рЃАрЃљрЃћрЃарЃЌрЃЮ рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљрЃА.

##### 2. рЃцрЃЮрЃарЃЏрЃљрЃбрЃўрЃА рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃљ

```php
class CSVReport extends ReportGenerator {
    protected function fetchData() {
        return [
            ["ID", "Name", "Salary"],
            [1, "John Doe", 5000],
            [2, "Jane Smith", 6000]
        ];
    }

    protected function formatData($data) {
        $csv = "";
        foreach ($data as $row) {
            $csv .= implode(",", $row) . "\n";
        }
        return $csv;
    }

    protected function exportReport($formattedData) {
        file_put_contents("report.csv", $formattedData);
        echo "CSV рЃљрЃюрЃњрЃљрЃарЃўрЃерЃў рЃерЃћрЃЦрЃЏрЃюрЃўрЃџрЃўрЃљ!\n";
    }
}
```

##### 3. HTML рЃарЃћрЃърЃЮрЃарЃбрЃўрЃА рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃљ

```php
class HTMLReport extends ReportGenerator {
    protected function fetchData() {
        return [
            ["ID" => 1, "Name" => "John Doe", "Salary" => 5000],
            ["ID" => 2, "Name" => "Jane Smith", "Salary" => 6000]
        ];
    }

    protected function formatData($data) {
        $html = "<table border='1'><tr><th>ID</th><th>Name</th><th>Salary</th></tr>";
        foreach ($data as $row) {
            $html .= "<tr><td>{$row['ID']}</td><td>{$row['Name']}</td><td>{$row['Salary']}</td></tr>";
        }
        $html .= "</table>";
        return $html;
    }

    protected function exportReport($formattedData) {
        file_put_contents("report.html", $formattedData);
        echo "HTML рЃљрЃюрЃњрЃљрЃарЃўрЃерЃў рЃерЃћрЃЦрЃЏрЃюрЃўрЃџрЃўрЃљ!\n";
    }
}
```


##### 4. Template Method - рЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ


```php
$csvReport = new CSVReport();
$csvReport->generateReport(); // рЃЦрЃЏрЃюрЃўрЃА report.csv рЃцрЃљрЃўрЃџрЃА

$htmlReport = new HTMLReport();
$htmlReport->generateReport(); // рЃЦрЃЏрЃюрЃўрЃА report.html рЃцрЃљрЃўрЃџрЃА
```

##### 5. рЃерЃћрЃЊрЃћрЃњрЃў

```
CSV рЃљрЃюрЃњрЃљрЃарЃўрЃерЃў рЃерЃћрЃЦрЃЏрЃюрЃўрЃџрЃўрЃљ!
HTML рЃљрЃюрЃњрЃљрЃарЃўрЃерЃў рЃерЃћрЃЦрЃЏрЃюрЃўрЃџрЃўрЃљ!
```

#### Template Method рЃўрЃА рЃБрЃърЃўрЃарЃљрЃбрЃћрЃАрЃЮрЃЉрЃћрЃЉрЃў

РюЁ **рЃЎрЃЮрЃЊрЃўрЃА рЃњрЃљрЃюрЃЏрЃћрЃЮрЃарЃћрЃЉрЃўрЃА рЃерЃћрЃЏрЃфрЃўрЃарЃћрЃЉрЃљ** - рЃљрЃџрЃњрЃЮрЃарЃўрЃЌрЃЏрЃўрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљ рЃћрЃарЃЌ рЃљрЃЊрЃњрЃўрЃџрЃќрЃћ рЃўрЃюрЃљрЃ«рЃћрЃЉрЃљ.    
РюЁ **рЃњрЃљрЃЊрЃљрЃЏрЃгрЃДрЃЋрЃћрЃбрЃў рЃюрЃљрЃЉрЃўрЃ»рЃћрЃЉрЃўрЃА рЃЎрЃЮрЃюрЃцрЃўрЃњрЃБрЃарЃўрЃарЃћрЃЉрЃљрЃЊрЃЮрЃЉрЃљ** - рЃерЃЋрЃўрЃџрЃЮрЃЉрЃўрЃџ рЃЎрЃџрЃљрЃАрЃћрЃЉрЃА рЃерЃћрЃБрЃФрЃџрЃћрЃљрЃЌ рЃњрЃљрЃЊрЃљрЃгрЃћрЃарЃЮрЃю (Override) рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџрЃў рЃюрЃљрЃЉрЃўрЃ»рЃћрЃЉрЃў.    
РюЁ **рЃЎрЃЮрЃЊрЃўрЃА рЃЏрЃЮрЃЦрЃюрЃўрЃџрЃЮрЃЉрЃљ** - рЃљрЃ«рЃљрЃџрЃў рЃљрЃюрЃњрЃљрЃарЃўрЃерЃўрЃА рЃцрЃЮрЃарЃЏрЃљрЃбрЃўрЃА рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ рЃерЃћрЃАрЃљрЃФрЃџрЃћрЃЉрЃћрЃџрЃўрЃљ рЃЏрЃ«рЃЮрЃџрЃЮрЃЊ рЃљрЃ«рЃљрЃџрЃў рЃЎрЃџрЃљрЃАрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃўрЃЌ, рЃљрЃарЃАрЃћрЃЉрЃБрЃџрЃў рЃџрЃЮрЃњрЃўрЃЎрЃўрЃА рЃфрЃЋрЃџрЃўрЃџрЃћрЃЉрЃўрЃА рЃњрЃљрЃарЃћрЃерЃћ.

#### ­Ъџђ рЃЊрЃљрЃАрЃЎрЃЋрЃюрЃљ

Template Method рЃърЃљрЃбрЃћрЃарЃюрЃў рЃњрЃљрЃюрЃАрЃљрЃЎрЃБрЃЌрЃарЃћрЃЉрЃўрЃЌ рЃњрЃљрЃЏрЃЮрЃАрЃљрЃЊрЃћрЃњрЃўрЃљ, рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃњрЃЋрЃАрЃБрЃарЃА, рЃарЃЮрЃЏ рЃљрЃџрЃњрЃЮрЃарЃўрЃЌрЃЏрЃўрЃА рЃФрЃўрЃарЃўрЃЌрЃљрЃЊрЃў рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљ рЃБрЃфрЃЋрЃџрЃћрЃџрЃў рЃЊрЃљрЃарЃЕрЃћрЃА, рЃ«рЃЮрЃџрЃЮ рЃЏрЃўрЃАрЃў рЃфрЃљрЃџрЃЎрЃћрЃБрЃџрЃў рЃћрЃбрЃљрЃърЃћрЃЉрЃў рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџ рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃљрЃерЃў рЃњрЃљрЃюрЃАрЃ«рЃЋрЃљрЃЋрЃЊрЃћрЃЉрЃЮрЃЊрЃћрЃА. рЃЏрЃўрЃАрЃў рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ рЃЏрЃљрЃарЃбрЃўрЃЋрЃўрЃљ рЃЊрЃљ рЃЎрЃЮрЃЊрЃўрЃА рЃЮрЃарЃњрЃљрЃюрЃўрЃќрЃћрЃЉрЃБрЃџрЃЮрЃЉрЃљрЃА рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЋрЃюрЃљрЃЊ рЃљрЃБрЃЏрЃ»рЃЮрЃЉрЃћрЃАрЃћрЃЉрЃА.
