
**Strategy** рЃърЃљрЃбрЃћрЃарЃюрЃў рЃћрЃарЃЌ-рЃћрЃарЃЌрЃў рЃДрЃЋрЃћрЃџрЃљрЃќрЃћ рЃњрЃљрЃЋрЃарЃфрЃћрЃџрЃћрЃЉрЃБрЃџрЃў рЃЦрЃфрЃћрЃЋрЃўрЃЌрЃў рЃбрЃўрЃърЃўрЃА рЃЊрЃўрЃќрЃљрЃўрЃюрЃўрЃА рЃърЃљрЃбрЃћрЃарЃюрЃўрЃљ рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃљрЃА рЃњрЃЋрЃљрЃФрЃџрЃћрЃЋрЃА, рЃЊрЃўрЃюрЃљрЃЏрЃўрЃБрЃарЃљрЃЊ рЃерЃћрЃЋрЃфрЃЋрЃљрЃџрЃЮрЃЌ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃА рЃЦрЃфрЃћрЃЋрЃљ рЃњрЃљрЃерЃЋрЃћрЃЉрЃўрЃА рЃЊрЃарЃЮрЃА.

рЃћрЃА рЃърЃљрЃбрЃћрЃарЃюрЃў рЃњрЃљрЃюрЃАрЃљрЃЎрЃБрЃЌрЃарЃћрЃЉрЃўрЃЌ рЃњрЃљрЃЏрЃЮрЃАрЃљрЃЊрЃћрЃњрЃўрЃљ, рЃарЃЮрЃфрЃљ рЃњрЃЋрЃГрЃўрЃарЃЊрЃћрЃЉрЃљ рЃћрЃарЃЌрЃў рЃЊрЃљрЃўрЃњрЃўрЃЋрЃћ рЃљрЃџрЃњрЃЮрЃарЃўрЃЌрЃЏрЃўрЃА рЃАрЃ«рЃЋрЃљрЃЊрЃљрЃАрЃ«рЃЋрЃљ рЃарЃћрЃљрЃџрЃўрЃќрЃљрЃфрЃўрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃЌрЃљ рЃњрЃљрЃЊрЃљрЃарЃЌрЃЋрЃљ рЃерЃћрЃАрЃљрЃФрЃџрЃћрЃЉрЃћрЃџрЃўрЃљ рЃЎрЃЮрЃЊрЃўрЃА рЃЎрЃБрЃЉрЃџрЃўрЃарЃћрЃЉрЃўрЃА рЃњрЃљрЃарЃћрЃерЃћ.

---

### рЃърЃарЃЮрЃЉрЃџрЃћрЃЏрЃўрЃА рЃљрЃдрЃгрЃћрЃарЃљ
рЃЊрЃљрЃЋрЃБрЃерЃЋрЃљрЃЌ, рЃњрЃЋрЃљрЃЦрЃЋрЃА рЃњрЃљрЃЊрЃљрЃ«рЃЊрЃўрЃА рЃАрЃўрЃАрЃбрЃћрЃЏрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃАрЃ«рЃЋрЃљрЃЊрЃљрЃАрЃ«рЃЋрЃљ рЃЏрЃћрЃЌрЃЮрЃЊрЃА (PayPal, рЃАрЃљрЃЉрЃљрЃюрЃЎрЃЮ рЃЉрЃљрЃарЃљрЃЌрЃў, рЃЎрЃарЃўрЃърЃбрЃЮрЃЋрЃљрЃџрЃБрЃбрЃљ рЃЊрЃљ рЃљ.рЃе) рЃўрЃДрЃћрЃюрЃћрЃЉрЃА.
рЃбрЃарЃљрЃЊрЃўрЃфрЃўрЃБрЃџрЃў рЃЏрЃўрЃЊрЃњрЃЮрЃЏрЃўрЃЌ, рЃерЃћрЃўрЃФрЃџрЃћрЃЉрЃљ рЃњрЃљрЃЊрЃљрЃ«рЃЊрЃўрЃА рЃџрЃЮрЃњрЃўрЃЎрЃљ рЃЕрЃљрЃЋрЃгрЃћрЃарЃЮрЃЌ рЃЊрЃўрЃЊ if-else рЃЉрЃџрЃЮрЃЎрЃерЃў, рЃарЃљрЃф рЃЎрЃЮрЃЊрЃўрЃА рЃгрЃљрЃЎрЃўрЃЌрЃ«рЃЋрЃљрЃА рЃЊрЃљ рЃњрЃљрЃцрЃљрЃарЃЌрЃЮрЃћрЃЉрЃљрЃА рЃњрЃљрЃљрЃарЃЌрЃБрЃџрЃћрЃЉрЃА.
```php
class PaymentProcessor {
    public function pay($method, $amount) {
        if ($method === 'paypal') {
            echo "рЃњрЃљрЃЊрЃљрЃ«рЃЊрЃљ PayPal-рЃўрЃЌ: $amount USD";
        } elseif ($method === 'credit_card') {
            echo "рЃњрЃљрЃЊрЃљрЃ«рЃЊрЃљ рЃАрЃљрЃЉрЃљрЃюрЃЎрЃЮ рЃЉрЃљрЃарЃљрЃЌрЃўрЃЌ: $amount USD";
        } else {
            echo "рЃБрЃфрЃюрЃЮрЃЉрЃў рЃњрЃљрЃЊрЃљрЃ«рЃЊрЃўрЃА рЃЏрЃћрЃЌрЃЮрЃЊрЃў!";
        }
    }
}

$processor = new PaymentProcessor();
$processor->pay('paypal', 100);
```

рЃљрЃЦ рЃърЃарЃЮрЃЉрЃџрЃћрЃЏрЃљ рЃўрЃА рЃљрЃарЃўрЃА, рЃарЃЮрЃЏ рЃДрЃЮрЃЋрЃћрЃџрЃў рЃљрЃ«рЃљрЃџрЃў рЃњрЃљрЃЊрЃљрЃ«рЃЊрЃўрЃА рЃЏрЃћрЃЌрЃЮрЃЊрЃўрЃА рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ рЃЎрЃЮрЃЊрЃўрЃА рЃфрЃЋрЃџрЃўрЃџрЃћрЃЉрЃљрЃА рЃЏрЃЮрЃўрЃЌрЃ«рЃЮрЃЋрЃА, рЃарЃљрЃф рЃљрЃарЃдрЃЋрЃћрЃЋрЃА. **Open/Closed рЃърЃарЃўрЃюрЃфрЃўрЃърЃА**.

---

#### Strategy рЃърЃљрЃбрЃћрЃарЃюрЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ

рЃљрЃЏ рЃърЃарЃЮрЃЉрЃџрЃћрЃЏрЃўрЃА рЃњрЃљрЃЊрЃљрЃГрЃарЃўрЃАрЃЌрЃЋрЃўрЃА рЃњрЃљрЃЏрЃЮрЃЋрЃўрЃДрЃћрЃюрЃЮрЃЌ Strategy рЃърЃљрЃбрЃћрЃарЃюрЃў рЃЊрЃљ рЃЌрЃўрЃЌрЃўрЃЮрЃћрЃБрЃџрЃў рЃњрЃљрЃЊрЃљрЃ«рЃЊрЃўрЃА рЃЏрЃћрЃЌрЃЮрЃЊрЃў рЃфрЃљрЃџрЃЎрЃћ рЃЎрЃџрЃљрЃАрЃљрЃЊ рЃњрЃљрЃЏрЃЮрЃЋрЃДрЃЮрЃЌ.

##### 1. рЃЋрЃЦрЃЏрЃюрЃўрЃЌ strategy рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃА

```php
interface PaymentStrategy {
    public function pay($amount);
}
```

##### 2. рЃЋрЃЦрЃЏрЃюрЃўрЃЌ рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџ рЃАрЃбрЃарЃљрЃбрЃћрЃњрЃўрЃћрЃЉрЃА (рЃњрЃљрЃЊрЃљрЃ«рЃЊрЃўрЃА рЃЏрЃћрЃЌрЃЮрЃЊрЃћрЃЉрЃА)

```php
class PayPalPayment implements PaymentStrategy {
    public function pay($amount) {
        echo "рЃњрЃљрЃЊрЃљрЃ«рЃЊрЃљ PayPal-рЃўрЃЌ: $amount USD";
    }
}

class CreditCardPayment implements PaymentStrategy {
    public function pay($amount) {
        echo "рЃњрЃљрЃЊрЃљрЃ«рЃЊрЃљ рЃАрЃљрЃЉрЃљрЃюрЃЎрЃЮ рЃЉрЃљрЃарЃљрЃЌрЃўрЃЌ: $amount USD";
    }
}

class CryptoPayment implements PaymentStrategy {
    public function pay($amount) {
        echo "рЃњрЃљрЃЊрЃљрЃ«рЃЊрЃљ рЃЎрЃарЃўрЃърЃбрЃЮрЃЋрЃљрЃџрЃБрЃбрЃўрЃЌ: $amount USD";
    }
}
```

##### 3. рЃЋрЃЦрЃЏрЃюрЃўрЃЌ рЃЎрЃЮрЃюрЃбрЃћрЃЦрЃАрЃб рЃЎрЃџрЃљрЃАрЃА, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃАрЃбрЃарЃљрЃбрЃћрЃњрЃўрЃљрЃА рЃўрЃДрЃћрЃюрЃћрЃЉрЃА
```php
class PaymentContext {
    private $paymentStrategy;

    public function __construct(PaymentStrategy $paymentStrategy) {
        $this->paymentStrategy = $paymentStrategy;
    }

    public function executePayment($amount) {
        $this->paymentStrategy->pay($amount);
    }
}
```


##### 4. Strategy рЃърЃљрЃбрЃћрЃарЃюрЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ рЃЎрЃЮрЃЊрЃерЃў
```php
$paypalPayment = new PaymentContext(new PayPalPayment());
$paypalPayment->executePayment(100);

$cardPayment = new PaymentContext(new CreditCardPayment());
$cardPayment->executePayment(200);

$cryptoPayment = new PaymentContext(new CryptoPayment());
$cryptoPayment->executePayment(300);
```

```php
рЃњрЃљрЃЊрЃљрЃ«рЃЊрЃљ рЃЎрЃарЃўрЃърЃбрЃЮрЃЋрЃљрЃџрЃБрЃбрЃўрЃЌ: 300 USD                                                                                    
```

#### рЃБрЃърЃўрЃарЃљрЃбрЃћрЃАрЃЮрЃЉрЃћрЃЉрЃў

Strategy рЃърЃљрЃбрЃћрЃарЃюрЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃўрЃА рЃерЃћрЃЏрЃЊрЃћрЃњ:
- **рЃЎрЃЮрЃЊ8 рЃБрЃцрЃарЃЮ рЃЏрЃЮрЃгрЃћрЃАрЃарЃўрЃњрЃћрЃЉрЃБрЃџрЃў рЃЊрЃљ рЃњрЃљрЃцрЃљрЃарЃЌрЃЮрЃћрЃЉрЃљрЃЊрЃў рЃњрЃљрЃ«рЃЊрЃљ** - рЃљрЃ«рЃљрЃџрЃў рЃњрЃљрЃЊрЃљрЃ«рЃЊрЃўрЃА рЃЏрЃћрЃЌрЃЮрЃЊрЃўрЃА рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ рЃерЃћрЃАрЃљрЃФрЃџрЃћрЃЉрЃћрЃџрЃўрЃљ рЃФрЃўрЃарЃўрЃЌрЃљрЃЊрЃў рЃЎрЃџрЃљрЃАрЃўрЃА рЃфрЃЋрЃџрЃўрЃџрЃћрЃЉрЃўрЃА рЃњрЃљрЃарЃћрЃерЃћ.
- **рЃЏрЃЮрЃЊрЃБрЃџрЃљрЃарЃЮрЃЉрЃљ** - рЃЌрЃўрЃЌрЃЮрЃћрЃБрЃџрЃў рЃњрЃљрЃЊрЃљрЃ«рЃЊрЃўрЃА рЃЏрЃћрЃЌрЃЮрЃЊрЃў рЃЊрЃљрЃЏрЃЮрЃБрЃЎрЃўрЃЊрЃћрЃЉрЃћрЃџ рЃЎрЃџрЃљрЃАрЃћрЃЉрЃерЃўрЃљ рЃЊрЃљ рЃљрЃЊрЃЋрЃўрЃџрЃљрЃЊ рЃњрЃљрЃЏрЃЮрЃАрЃљрЃфрЃЋрЃџрЃћрЃџрЃўрЃљ.
- **рЃЎрЃЮрЃЊрЃўрЃА рЃњрЃљрЃЏрЃљрЃарЃбрЃўрЃЋрЃћрЃЉрЃљ** - рЃЌрЃљрЃЋрЃўрЃЊрЃљрЃю рЃљрЃЋрЃўрЃфрЃўрЃџрЃћрЃЌ if-else рЃЉрЃџрЃЮрЃЎрЃћрЃЉрЃў.

рЃљрЃЏ рЃърЃљрЃбрЃћрЃарЃюрЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ рЃљрЃарЃљ рЃЏрЃ«рЃЮрЃџрЃЮрЃЊ рЃњрЃљрЃЊрЃљрЃ«рЃЊрЃўрЃА рЃАрЃўрЃАрЃбрЃћрЃЏрЃћрЃЉрЃерЃў, рЃљрЃарЃљрЃЏрЃћрЃЊ рЃюрЃћрЃЉрЃўрЃАрЃЏрЃўрЃћрЃа рЃърЃарЃЮрЃћрЃЦрЃбрЃерЃў рЃерЃћрЃўрЃФрЃџрЃћрЃЉрЃљ, рЃАрЃљрЃЊрЃљрЃф рЃАрЃљрЃГрЃўрЃарЃЮрЃљ рЃЏрЃЮрЃЦрЃюрЃўрЃџрЃў рЃљрЃџрЃњрЃЮрЃарЃўрЃЌрЃЏрЃћрЃЉрЃўрЃА рЃЏрЃљрЃарЃЌрЃЋрЃљ, рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃљрЃЊ, рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃЌрЃљ рЃАрЃЮрЃарЃбрЃўрЃарЃћрЃЉрЃљ, рЃџрЃЮрЃњрЃўрЃарЃћрЃЉрЃўрЃА рЃАрЃўрЃАрЃбрЃћрЃЏрЃљ рЃЊрЃљ рЃљ.рЃе

---

#### ­Ъџђ рЃЊрЃљрЃАрЃЎрЃЋрЃюрЃљ

**Strategy** рЃърЃљрЃбрЃћрЃарЃюрЃў рЃљрЃарЃўрЃА рЃФрЃџрЃўрЃћрЃарЃў рЃўрЃюрЃАрЃбрЃарЃБрЃЏрЃћрЃюрЃбрЃў, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃўрЃАрЃбрЃћрЃЉрЃА рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃљрЃА рЃљрЃФрЃџрЃћрЃЋрЃА, рЃАрЃўрЃАрЃбрЃћрЃЏрЃљрЃбрЃБрЃарЃљрЃЊ рЃЏрЃљрЃарЃЌрЃЮрЃю рЃАрЃ«рЃЋрЃљрЃЊрЃљрЃАрЃ«рЃЋрЃљ рЃЦрЃфрЃћрЃЋрЃљ, рЃЊрЃљрЃўрЃфрЃЋрЃљрЃџ SOLID рЃърЃарЃўрЃюрЃфрЃўрЃърЃћрЃЉрЃў рЃЊрЃљ рЃЎрЃЮрЃЊрЃў рЃБрЃцрЃарЃЮ рЃЏрЃЮрЃЊрЃБрЃџрЃљрЃарЃБрЃџрЃў рЃЊрЃљ рЃљрЃЊрЃЋрЃўрЃџрЃљрЃЊ рЃерЃћрЃАрЃљрЃюрЃљрЃарЃЕрЃБрЃюрЃћрЃЉрЃћрЃџрЃў рЃњрЃљрЃ«рЃљрЃЊрЃЮрЃю.
