**Observer (рЃЊрЃљрЃЏрЃЎрЃЋрЃўрЃарЃЋрЃћрЃЉрЃћрЃџрЃў)** рЃљрЃарЃўрЃА рЃЦрЃфрЃћрЃЋрЃўрЃЌрЃў рЃбрЃўрЃърЃўрЃА рЃърЃљрЃбрЃћрЃарЃюрЃў, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћрЃЉрЃљ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃА рЃерЃЮрЃарЃўрЃА рЃЊрЃљрЃЏрЃЮрЃЎрЃўрЃЊрЃћрЃЉрЃБрЃџрЃћрЃЉрЃўрЃА рЃАрЃљрЃЏрЃљрЃарЃЌрЃљрЃЋрЃљрЃЊ. рЃћрЃА рЃърЃљрЃбрЃћрЃарЃюрЃў рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃљрЃА рЃњрЃЋрЃљрЃФрЃџрЃћрЃЋрЃА, рЃарЃЮрЃЏ рЃћрЃарЃЌрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃА рЃфрЃЋрЃџрЃўрЃџрЃћрЃЉрЃћрЃЉрЃў рЃљрЃЋрЃбрЃЮрЃЏрЃљрЃбрЃБрЃарЃљрЃЊ рЃљрЃўрЃАрЃљрЃ«рЃЮрЃА рЃЏрЃўрЃА рЃЊрЃљрЃЏрЃЎрЃЋрЃўрЃарЃЋрЃћрЃЉрЃћрЃџрЃќрЃћ рЃўрЃАрЃћ, рЃарЃЮрЃЏ рЃЏрЃљрЃЌ рЃерЃЮрЃарЃўрЃА рЃърЃўрЃарЃЊрЃљрЃърЃўрЃарЃў рЃЊрЃљрЃЏрЃЮрЃЎрЃўрЃЊрЃћрЃЉрЃБрЃџрЃћрЃЉрЃљ рЃљрЃа рЃљрЃарЃАрЃћрЃЉрЃЮрЃЉрЃЊрЃћрЃА.

#### рЃарЃЮрЃњрЃЮрЃа рЃњрЃљрЃЏрЃЮрЃЋрЃўрЃДрЃћрЃюрЃЮрЃЌ рЃћрЃА рЃърЃљрЃбрЃћрЃарЃюрЃў?

Observer рЃърЃљрЃбрЃћрЃарЃюрЃў рЃњрЃљрЃюрЃАрЃљрЃЎрЃБрЃЌрЃарЃћрЃЉрЃўрЃЌ рЃАрЃљрЃАрЃљрЃарЃњрЃћрЃЉрЃџрЃЮрЃљрЃЊ, рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃњрЃЋрЃГрЃўрЃарЃЊрЃћрЃЉрЃљ:
- рЃћрЃарЃЌрЃў рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃА **рЃфрЃЋрЃџрЃўрЃџрЃћрЃЉрЃћрЃЉрЃќрЃћ** рЃарЃћрЃљрЃњрЃћрЃўрЃарЃћрЃЉрЃљ рЃАрЃ«рЃЋрЃљ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃЏрЃўрЃћрЃа.
- **Loose Coupling (рЃАрЃБрЃАрЃбрЃў рЃЎрЃљрЃЋрЃерЃўрЃарЃў)** рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃА рЃерЃЮрЃарЃўрЃЊ, рЃарЃљрЃЌрЃљ рЃфрЃЋрЃџрЃўрЃџрЃћрЃЉрЃћрЃЉрЃў рЃЏрЃљрЃарЃбрЃўрЃЋрЃљрЃЊ рЃЏрЃЮрЃ«рЃЊрЃћрЃА.
- **рЃўрЃЋрЃћрЃюрЃЌрЃћрЃЉрЃўрЃА** рЃЏрЃљрЃарЃЌрЃЋрЃљ рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљрЃерЃў.

#### Observer рЃърЃљрЃбрЃћрЃарЃюрЃўрЃА рЃўрЃЏрЃърЃџрЃћрЃЏрЃћрЃюрЃбрЃљрЃфрЃўрЃљ

рЃЦрЃЋрЃћрЃЋрЃўрЃЌ рЃЏрЃЮрЃфрЃћрЃЏрЃБрЃџрЃўрЃљ рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў, рЃЌрЃБ рЃарЃЮрЃњрЃЮрЃа рЃерЃћрЃњрЃЋрЃўрЃФрЃџрЃўрЃљ рЃљрЃЏ рЃърЃљрЃбрЃћрЃарЃюрЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ.

##### 1. Subject рЃЎрЃџрЃљрЃАрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ

рЃћрЃА рЃЎрЃџрЃљрЃАрЃў рЃўрЃюрЃљрЃ«рЃљрЃЋрЃА рЃЊрЃљрЃЏрЃЎрЃЋрЃўрЃарЃЋрЃћрЃЉрЃџрЃћрЃЉрЃўрЃА рЃАрЃўрЃљрЃА рЃЊрЃљ рЃЏрЃљрЃЌ рЃћрЃфрЃюрЃЮрЃЉрЃћрЃЉрЃљ, рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃў рЃўрЃфрЃЋрЃџрЃћрЃЉрЃљ.
```php
<?php

interface Subject {
    public function attach(Observer $observer);
    public function detach(Observer $observer);
    public function notify();
}

class NewsPublisher implements Subject {
    private array $observers = [];
    private string $latestNews;

    public function attach(Observer $observer) {
        $this->observers[] = $observer;
    }

    public function detach(Observer $observer) {
        $this->observers = array_filter($this->observers, fn($obs) => $obs !== $observer);
    }

    public function notify() {
        foreach ($this->observers as $observer) {
            $observer->update($this->latestNews);
        }
    }

    public function setNews(string $news) {
        $this->latestNews = $news;
        $this->notify(); // рЃДрЃЋрЃћрЃџрЃљ рЃЊрЃљрЃЏрЃЎрЃЋрЃўрЃарЃЋрЃћрЃЉрЃћрЃџрЃў рЃЏрЃўрЃўрЃдрЃћрЃЉрЃА рЃњрЃљрЃюрЃљрЃ«рЃџрЃћрЃЉрЃБрЃџ рЃАрЃўрЃљрЃ«рЃџрЃћрЃА
    }
}
```

##### 2. Observar (рЃЊрЃљрЃЏрЃЎрЃЋрЃўрЃарЃЋрЃћрЃЉрЃћрЃџрЃў) рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃў рЃЊрЃљ рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџрЃў рЃЊрЃљрЃЏрЃЎрЃЋрЃўрЃарЃЋрЃћрЃЉрЃћрЃџрЃў

**Observer** рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃў рЃњрЃљрЃюрЃАрЃљрЃќрЃдрЃЋрЃарЃљрЃЋрЃА **update()** рЃЏрЃћрЃЌрЃЮрЃЊрЃА, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃњрЃљрЃюрЃљрЃ«рЃџрЃЊрЃћрЃЉрЃљ **subject -рЃўрЃА рЃфрЃЋрЃџрЃўрЃџрЃћрЃЉрЃўрЃАрЃљрЃА**.

```php
<?php

interface Observer {
    public function update(string $news);
}

class EmailSubscriber implements Observer {
    private string $name;

    public function __construct(string $name) {
        $this->name = $name;
    }

    public function update(string $news) {
        echo "­ЪЊД {$this->name} рЃЏрЃўрЃўрЃдрЃЮ рЃљрЃ«рЃљрЃџрЃў рЃАрЃўрЃљрЃ«рЃџрЃћ: $news\n";
    }
}

class SmsSubscriber implements Observer {
    private string $phoneNumber;

    public function __construct(string $phoneNumber) {
        $this->phoneNumber = $phoneNumber;
    }

    public function update(string $news) {
        echo "­ЪЊ▓ SMS рЃњрЃљрЃњрЃќрЃљрЃЋрЃюрЃўрЃџрЃўрЃљ ({$this->phoneNumber}): $news\n";
    }
}
```
